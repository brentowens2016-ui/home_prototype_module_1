21/// Represents a Python exception.
22///
23/// To avoid needing access to [`Python`] in `Into` conversions to create `PyErr` (thus improving
24/// compatibility with `?` and other Rust errors) this type supports creating exceptions instances
25/// in a lazy fashion, where the full Python object for the exception is created only when needed.
26///
27/// Accessing the contained exception in any way, such as with [`value`](PyErr::value),
28/// [`get_type`](PyErr::get_type), or [`is_instance`](PyErr::is_instance) will create the full
29/// exception object if it was not already created.
30pub struct PyErr {
31    // Safety: can only hand out references when in the "normalized" state. Will never change
32    // after normalization.
33    //
34    // The state is temporarily removed from the PyErr during normalization, to avoid
35    // concurrent modifications.
36    state: UnsafeCell<Option<PyErrState>>,
37}
38
39// The inner value is only accessed through ways that require proving the gil is held
40#[cfg(feature = "nightly")]
41unsafe impl crate::marker::Ungil for PyErr {}
42unsafe impl Send for PyErr {}
43unsafe impl Sync for PyErr {}
44

46pub type PyResult<T> = Result<T, PyErr>;

129    /// Creates a new PyErr of type `T`.
130    ///
131    /// `args` can be:
132    /// * a tuple: the exception instance will be created using the equivalent to the Python
133    ///   expression `T(*tuple)`
134    /// * any other value: the exception instance will be created using the equivalent to the Python
135    ///   expression `T(value)`
136    ///
137    /// This exception instance will be initialized lazily. This avoids the need for the Python GIL
138    /// to be held, but requires `args` to be `Send` and `Sync`. If `args` is not `Send` or `Sync`,
139    /// consider using [`PyErr::from_value`] instead.
140    ///
141    /// If `T` does not inherit from `BaseException`, then a `TypeError` will be returned.
142    ///
143    /// If calling T's constructor with `args` raises an exception, that exception will be returned.
144    ///
145    /// # Examples
146    ///
147    /// ```
148    /// use pyo3::prelude::*;
149    /// use pyo3::exceptions::PyTypeError;
150    ///
151    /// #[pyfunction]
152    /// fn always_throws() -> PyResult<()> {
153    ///     Err(PyErr::new::<PyTypeError, _>("Error message"))
154    /// }
155    /// #
156    /// # Python::with_gil(|py| {
157    /// #     let fun = pyo3::wrap_pyfunction_bound!(always_throws, py).unwrap();
158    /// #     let err = fun.call0().expect_err("called a function that should always return an error but the return value was Ok");
159    /// #     assert!(err.is_instance_of::<PyTypeError>(py))
160    /// # });
161    /// ```
162    ///
163    /// In most cases, you can use a concrete exception's constructor instead:
164    ///
165    /// ```
166    /// use pyo3::prelude::*;
167    /// use pyo3::exceptions::PyTypeError;
168    ///
169    /// #[pyfunction]
170    /// fn always_throws() -> PyResult<()> {
171    ///     Err(PyTypeError::new_err("Error message"))
172    /// }
173    /// #
174    /// # Python::with_gil(|py| {
175    /// #     let fun = pyo3::wrap_pyfunction_bound!(always_throws, py).unwrap();
176    /// #     let err = fun.call0().expect_err("called a function that should always return an error but the return value was Ok");
177    /// #     assert!(err.is_instance_of::<PyTypeError>(py))
178    /// # });
179    /// ```
180    #[inline]
181    pub fn new<T, A>(args: A) -> PyErr
182    where
183        T: PyTypeInfo,
184        A: PyErrArguments + Send + Sync + 'static,
185    {
186        PyErr::from_state(PyErrState::Lazy(Box::new(move |py| {
187            PyErrStateLazyFnOutput {
188                ptype: T::type_object_bound(py).into(),
189                pvalue: args.arguments(py),
190            }
191        })))
192    }

238    /// Creates a new PyErr.
239    ///
240    /// If `obj` is a Python exception object, the PyErr will contain that object.
241    ///
242    /// If `obj` is a Python exception type object, this is equivalent to `PyErr::from_type(obj, ())`.
243    ///
244    /// Otherwise, a `TypeError` is created.
245    ///
246    /// # Examples
247    /// ```rust
248    /// use pyo3::prelude::*;
249    /// use pyo3::PyTypeInfo;
250    /// use pyo3::exceptions::PyTypeError;
251    /// use pyo3::types::PyString;
252    ///
253    /// Python::with_gil(|py| {
254    ///     // Case #1: Exception object
255    ///     let err = PyErr::from_value_bound(PyTypeError::new_err("some type error")
256    ///         .value_bound(py).clone().into_any());
257    ///     assert_eq!(err.to_string(), "TypeError: some type error");
258    ///
259    ///     // Case #2: Exception type
260    ///     let err = PyErr::from_value_bound(PyTypeError::type_object_bound(py).into_any());
261    ///     assert_eq!(err.to_string(), "TypeError: ");
262    ///
263    ///     // Case #3: Invalid exception value
264    ///     let err = PyErr::from_value_bound(PyString::new_bound(py, "foo").into_any());
265    ///     assert_eq!(
266    ///         err.to_string(),
267    ///         "TypeError: exceptions must derive from BaseException"
268    ///     );
269    /// });
270    /// ```
271    pub fn from_value_bound(obj: Bound<'_, PyAny>) -> PyErr {
272        let state = match obj.downcast_into::<PyBaseException>() {
273            Ok(obj) => PyErrState::normalized(obj),
274            Err(err) => {
275                // Assume obj is Type[Exception]; let later normalization handle if this
276                // is not the case
277                let obj = err.into_inner();
278                let py = obj.py();
279                PyErrState::lazy(obj.into_py(py), py.None())
280            }
281        };
282
283        PyErr::from_state(state)
284    }

390    /// Takes the current error from the Python interpreter's global state and clears the global
391    /// state. If no error is set, returns `None`.
392    ///
393    /// If the error is a `PanicException` (which would have originated from a panic in a pyo3
394    /// callback) then this function will resume the panic.
395    ///
396    /// Use this function when it is not known if an error should be present. If the error is
397    /// expected to have been set, for example from [`PyErr::occurred`] or by an error return value
398    /// from a C FFI function, use [`PyErr::fetch`].
399    pub fn take(py: Python<'_>) -> Option<PyErr> {
400        Self::_take(py)
401    }

403    #[cfg(not(Py_3_12))]
404    fn _take(py: Python<'_>) -> Option<PyErr> {
405        let (ptype, pvalue, ptraceback) = unsafe {
406            let mut ptype: *mut ffi::PyObject = std::ptr::null_mut();
407            let mut pvalue: *mut ffi::PyObject = std::ptr::null_mut();
408            let mut ptraceback: *mut ffi::PyObject = std::ptr::null_mut();
409            ffi::PyErr_Fetch(&mut ptype, &mut pvalue, &mut ptraceback);
410
411            // Convert to Py immediately so that any references are freed by early return.
412            let ptype = PyObject::from_owned_ptr_or_opt(py, ptype);
413            let pvalue = PyObject::from_owned_ptr_or_opt(py, pvalue);
414            let ptraceback = PyObject::from_owned_ptr_or_opt(py, ptraceback);
415
416            // A valid exception state should always have a non-null ptype, but the other two may be
417            // null.
418            let ptype = match ptype {
419                Some(ptype) => ptype,
420                None => {
421                    debug_assert!(
422                        pvalue.is_none(),
423                        "Exception type was null but value was not null"
424                    );
425                    debug_assert!(
426                        ptraceback.is_none(),
427                        "Exception type was null but traceback was not null"
428                    );
429                    return None;
430                }
431            };
432
433            (ptype, pvalue, ptraceback)
434        };
435
436        if ptype.as_ptr() == PanicException::type_object_raw(py).cast() {
437            let msg = pvalue
438                .as_ref()
439                .and_then(|obj| obj.bind(py).str().ok())
440                .map(|py_str| py_str.to_string_lossy().into())
441                .unwrap_or_else(|| String::from("Unwrapped panic from Python code"));
442
443            let state = PyErrState::FfiTuple {
444                ptype,
445                pvalue,
446                ptraceback,
447            };
448            Self::print_panic_and_unwind(py, state, msg)
449        }
450
451        Some(PyErr::from_state(PyErrState::FfiTuple {
452            ptype,
453            pvalue,
454            ptraceback,
455        }))
456    }

460        let state = PyErrStateNormalized::take(py)?;
461        let pvalue = state.pvalue.bind(py);
462        if pvalue.get_type().as_ptr() == PanicException::type_object_raw(py).cast() {
463            let msg: String = pvalue
464                .str()
465                .map(|py_str| py_str.to_string_lossy().into())
466                .unwrap_or_else(|_| String::from("Unwrapped panic from Python code"));
467            Self::print_panic_and_unwind(py, PyErrState::Normalized(state), msg)
468        }
469
470        Some(PyErr::from_state(PyErrState::Normalized(state)))
471    }

473    fn print_panic_and_unwind(py: Python<'_>, state: PyErrState, msg: String) -> ! {
474        eprintln!("--- PyO3 is resuming a panic after fetching a PanicException from Python. ---");
475        eprintln!("Python stack trace below:");

487    ///  - Panics in debug mode.
488    ///  - Returns a `SystemError` in release mode.
489    ///
490    /// This behavior is consistent with Python's internal handling of what happens when a C return
491    /// value indicates an error occurred but the global error state is empty. (A lack of exception
492    /// should be treated as a bug in the code which returned an error code but did not set an

500        const FAILED_TO_FETCH: &str = "attempted to fetch exception but none was set";
501        match PyErr::take(py) {
502            Some(err) => err,
503            #[cfg(debug_assertions)]
504            None => panic!("{}", FAILED_TO_FETCH),
505            #[cfg(not(debug_assertions))]
506            None => exceptions::PySystemError::new_err(FAILED_TO_FETCH),
507        }

534    /// Creates a new exception type with the given name and docstring.
535    ///
536    /// - `base` can be an existing exception type to subclass, or a tuple of classes.
537    /// - `dict` specifies an optional dictionary of class variables and methods.
538    /// - `doc` will be the docstring seen by python users.
539    ///
540    ///
541    /// # Errors
542    ///
543    /// This function returns an error if `name` is not of the form `<module>.<ExceptionName>`.
544    ///
545    /// # Panics
546    ///
547    /// This function will panic if  `name` or `doc` cannot be converted to [`CString`]s.
548    pub fn new_type_bound<'py>(
549        py: Python<'py>,
550        name: &str,
551        doc: Option<&str>,
552        base: Option<&Bound<'py, PyType>>,
553        dict: Option<PyObject>,
554    ) -> PyResult<Py<PyType>> {
555        let base: *mut ffi::PyObject = match base {
556            None => std::ptr::null_mut(),
557            Some(obj) => obj.as_ptr(),
558        };
559
560        let dict: *mut ffi::PyObject = match dict {
561            None => std::ptr::null_mut(),
562            Some(obj) => obj.as_ptr(),
563        };
564
565        let null_terminated_name =
566            CString::new(name).expect("Failed to initialize nul terminated exception name");
567
568        let null_terminated_doc =
569            doc.map(|d| CString::new(d).expect("Failed to initialize nul terminated docstring"));
570
571        let null_terminated_doc_ptr = match null_terminated_doc.as_ref() {
572            Some(c) => c.as_ptr(),
573            None => std::ptr::null(),
574        };
575
576        let ptr = unsafe {
577            ffi::PyErr_NewExceptionWithDoc(
578                null_terminated_name.as_ptr(),
579                null_terminated_doc_ptr,
580                base,
581                dict,
582            )
583        };
584
585        unsafe { Py::from_owned_ptr_or_err(py, ptr) }
586    }

606                traceback
607                    .as_ref()
608                    .map_or(std::ptr::null_mut(), |traceback| traceback.as_ptr()),
609            )
610        }

690    /// Reports the error as unraisable.
691    ///
692    /// This calls `sys.unraisablehook()` using the current exception and obj argument.
693    ///
694    /// This method is useful to report errors in situations where there is no good mechanism
695    /// to report back to the Python land.  In Python this is used to indicate errors in
696    /// background threads or destructors which are protected.  In Rust code this is commonly
697    /// useful when you are calling into a Python callback which might fail, but there is no
698    /// obvious way to handle this error other than logging it.
699    ///
700    /// Calling this method has the benefit that the error goes back into a standardized callback
701    /// in Python which for instance allows unittests to ensure that no unraisable error
702    /// actually happend by hooking `sys.unraisablehook`.
703    ///
704    /// Example:
705    /// ```rust
706    /// # use pyo3::prelude::*;
707    /// # use pyo3::exceptions::PyRuntimeError;
708    /// # fn failing_function() -> PyResult<()> { Err(PyRuntimeError::new_err("foo")) }
709    /// # fn main() -> PyResult<()> {
710    /// Python::with_gil(|py| {
711    ///     match failing_function() {
712    ///         Err(pyerr) => pyerr.write_unraisable_bound(py, None),
713    ///         Ok(..) => { /* do something here */ }
714    ///     }
715    ///     Ok(())
716    /// })
717    /// # }
718    #[inline]
719    pub fn write_unraisable_bound(self, py: Python<'_>, obj: Option<&Bound<'_, PyAny>>) {
720        self.restore(py);
721        unsafe { ffi::PyErr_WriteUnraisable(obj.map_or(std::ptr::null_mut(), Bound::as_ptr)) }
722    }

802    ///

804    ///

830            ffi::PyErr_WarnExplicit(
831                category.as_ptr(),
832                message.as_ptr(),
833                filename.as_ptr(),
834                lineno,
835                module_ptr,
836                registry,

838        })

842    ///

862    /// Return the cause (either an exception instance, or None, set by `raise ... from ...`)
863    /// associated with the exception, as accessible from Python through `__cause__`.
864    pub fn cause(&self, py: Python<'_>) -> Option<PyErr> {
865        use crate::ffi_ptr_ext::FfiPtrExt;
866        let obj = unsafe {
867            ffi::PyException_GetCause(self.value_bound(py).as_ptr()).assume_owned_or_opt(py)
868        };
869        // PyException_GetCause is documented as potentially returning PyNone, but only GraalPy seems to actually do that
870        #[cfg(GraalPy)]
871        if let Some(cause) = &obj {
872            if cause.is_none() {
873                return None;
874            }
875        }
876        obj.map(Self::from_value_bound)
877    }

879    /// Set the cause associated with the exception, pass `None` to clear it.
880    pub fn set_cause(&self, py: Python<'_>, cause: Option<Self>) {
881        let value = self.value_bound(py);
882        let cause = cause.map(|err| err.into_value(py));
883        unsafe {
884            // PyException_SetCause _steals_ a reference to cause, so must use .into_ptr()
885            ffi::PyException_SetCause(
886                value.as_ptr(),
887                cause.map_or(std::ptr::null_mut(), Py::into_ptr),
888            );
889        }
890    }

899    #[inline]
900    fn normalized(&self, py: Python<'_>) -> &PyErrStateNormalized {
901        if let Some(PyErrState::Normalized(n)) = unsafe {
902            // Safety: self.state will never be written again once normalized.
903            &*self.state.get()
904        } {
905            return n;
906        }
907
908        self.make_normalized(py)
909    }
910
911    #[cold]
912    fn make_normalized(&self, py: Python<'_>) -> &PyErrStateNormalized {
913        // This process is safe because:
914        // - Access is guaranteed not to be concurrent thanks to `Python` GIL token
915        // - Write happens only once, and then never will change again.
916        // - State is set to None during the normalization process, so that a second
917        //   concurrent normalization attempt will panic before changing anything.
918
919        let state = unsafe {
920            (*self.state.get())
921                .take()
922                .expect("Cannot normalize a PyErr while already normalizing it.")
923        };
924
925        unsafe {
926            let self_state = &mut *self.state.get();
927            *self_state = Some(PyErrState::Normalized(state.normalize(py)));
928            match self_state {
929                Some(PyErrState::Normalized(n)) => n,
930                _ => unreachable!(),
931            }
932        }
933    }

1324    fn warnings() {
1325        use crate::types::any::PyAnyMethods;
1326        // Note: although the warning filter is interpreter global, keeping the
1327        // GIL locked should prevent effects to be visible to other testing
1328        // threads.
1329        Python::with_gil(|py| {
1330            let cls = py.get_type_bound::<exceptions::PyUserWarning>();
1331
1332            // Reset warning filter to default state
1333            let warnings = py.import_bound("warnings").unwrap();
1334            warnings.call_method0("resetwarnings").unwrap();
1335
1336            // First, test the warning is emitted
1337            assert_warnings!(
1338                py,
1339                { PyErr::warn_bound(py, &cls, "I am warning you", 0).unwrap() },
1340                [(exceptions::PyUserWarning, "I am warning you")]
1341            );
1342
1343            // Test with raising
1344            warnings
1345                .call_method1("simplefilter", ("error", &cls))
1346                .unwrap();
1347            PyErr::warn_bound(py, &cls, "I am warning you", 0).unwrap_err();
1348
1349            // Test with error for an explicit module
1350            warnings.call_method0("resetwarnings").unwrap();
1351            warnings
1352                .call_method1("filterwarnings", ("error", "", &cls, "pyo3test"))
1353                .unwrap();
1354
1355            // This has the wrong module and will not raise, just be emitted
1356            assert_warnings!(
1357                py,
1358                { PyErr::warn_bound(py, &cls, "I am warning you", 0).unwrap() },
1359                [(exceptions::PyUserWarning, "I am warning you")]
1360            );
1361
1362            let err = PyErr::warn_explicit_bound(
1363                py,
1364                &cls,
1365                "I am warning you",
1366                "pyo3test.py",
1367                427,
1368                None,
1369                None,
1370            )
1371            .unwrap_err();

1373                .value_bound(py)
1374                .getattr("args")
1375                .unwrap()
1376                .get_item(0)
1377                .unwrap()
1378                .eq("I am warning you")

1381            // Finally, reset filter again
1382            warnings.call_method0("resetwarnings").unwrap();
1383        });
