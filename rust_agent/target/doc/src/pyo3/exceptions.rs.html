1//! Exception and warning types defined by Python.
2//!
3//! The structs in this module represent Python's built-in exceptions and
4//! warnings, while the modules comprise structs representing errors defined in
5//! Python code.
6//!
7//! The latter are created with the
8//! [`import_exception`](crate::import_exception) macro, which you can use
9//! yourself to import Python classes that are ultimately derived from
10//! `BaseException`.
11

17/// The boilerplate to convert between a Rust type and a Python exception.
18#[doc(hidden)]
19#[macro_export]
20macro_rules! impl_exception_boilerplate {
21    ($name: ident) => {
22        // FIXME https://github.com/PyO3/pyo3/issues/3903
23        #[allow(unknown_lints, non_local_definitions)]
24        impl ::std::convert::From<&$name> for $crate::PyErr {
25            #[inline]
26            fn from(err: &$name) -> $crate::PyErr {
27                #[allow(deprecated)]
28                $crate::PyErr::from_value(err)
29            }
30        }
31
32        $crate::impl_exception_boilerplate_bound!($name);
33
34        impl ::std::error::Error for $name {
35            fn source(&self) -> ::std::option::Option<&(dyn ::std::error::Error + 'static)> {
36                unsafe {
37                    #[allow(deprecated)]
38                    let cause: &$crate::exceptions::PyBaseException = self
39                        .py()
40                        .from_owned_ptr_or_opt($crate::ffi::PyException_GetCause(self.as_ptr()))?;
41
42                    ::std::option::Option::Some(cause)
43                }
44            }
45        }
46
47        impl $crate::ToPyErr for $name {}
48    };
49}

51#[doc(hidden)]
52#[macro_export]
53macro_rules! impl_exception_boilerplate_bound {
54    ($name: ident) => {
55        impl $name {
56            /// Creates a new [`PyErr`] of this type.
57            ///
58            /// [`PyErr`]: https://docs.rs/pyo3/latest/pyo3/struct.PyErr.html "PyErr in pyo3"
59            #[inline]
60            pub fn new_err<A>(args: A) -> $crate::PyErr
61            where
62                A: $crate::PyErrArguments + ::std::marker::Send + ::std::marker::Sync + 'static,
63            {
64                $crate::PyErr::new::<$name, A>(args)
65            }
66        }
67    };
68}

70/// Defines a Rust type for an exception defined in Python code.
71///
72/// # Syntax
73///
74/// ```import_exception!(module, MyError)```
75///
76/// * `module` is the name of the containing module.
77/// * `MyError` is the name of the new exception type.
78///
79/// # Examples
80/// ```
81/// use pyo3::import_exception;
82/// use pyo3::types::IntoPyDict;
83/// use pyo3::Python;
84///
85/// import_exception!(socket, gaierror);
86///
87/// Python::with_gil(|py| {
88///     let ctx = [("gaierror", py.get_type_bound::<gaierror>())].into_py_dict_bound(py);
89///     pyo3::py_run!(py, *ctx, "import socket; assert gaierror is socket.gaierror");
90/// });
91///
92/// ```
93#[macro_export]
94macro_rules! import_exception {
95    ($module: expr, $name: ident) => {
96        /// A Rust type representing an exception defined in Python code.
97        ///
98        /// This type was created by the [`pyo3::import_exception!`] macro - see its documentation
99        /// for more information.
100        ///
101        /// [`pyo3::import_exception!`]: https://docs.rs/pyo3/latest/pyo3/macro.import_exception.html "import_exception in pyo3"
102        #[repr(transparent)]
103        #[allow(non_camel_case_types)] // E.g. `socket.herror`
104        pub struct $name($crate::PyAny);
105
106        $crate::impl_exception_boilerplate!($name);
107
108        $crate::pyobject_native_type_core!(
109            $name,
110            $name::type_object_raw,
111            #module=::std::option::Option::Some(stringify!($module))
112        );
113
114        impl $name {
115            fn type_object_raw(py: $crate::Python<'_>) -> *mut $crate::ffi::PyTypeObject {
116                use $crate::types::PyTypeMethods;
117                static TYPE_OBJECT: $crate::impl_::exceptions::ImportedExceptionTypeObject =
118                    $crate::impl_::exceptions::ImportedExceptionTypeObject::new(stringify!($module), stringify!($name));
119                TYPE_OBJECT.get(py).as_type_ptr()
120            }
121        }
122    };
123}

125/// Variant of [`import_exception`](crate::import_exception) that does not emit code needed to
126/// use the imported exception type as a GIL Ref.
127///
128/// This is useful only during migration as a way to avoid generating needless code.
129#[macro_export]
130macro_rules! import_exception_bound {
131    ($module: expr, $name: ident) => {
132        /// A Rust type representing an exception defined in Python code.
133        ///
134        /// This type was created by the [`pyo3::import_exception_bound!`] macro - see its documentation
135        /// for more information.
136        ///
137        /// [`pyo3::import_exception_bound!`]: https://docs.rs/pyo3/latest/pyo3/macro.import_exception.html "import_exception in pyo3"
138        #[repr(transparent)]
139        #[allow(non_camel_case_types)] // E.g. `socket.herror`
140        pub struct $name($crate::PyAny);
141
142        $crate::impl_exception_boilerplate_bound!($name);
143
144        // FIXME remove this: was necessary while `PyTypeInfo` requires `HasPyGilRef`,
145        // should change in 0.22.
146        unsafe impl $crate::type_object::HasPyGilRef for $name {
147            type AsRefTarget = $crate::PyAny;
148        }
149
150        $crate::pyobject_native_type_info!(
151            $name,
152            $name::type_object_raw,
153            ::std::option::Option::Some(stringify!($module))
154        );
155
156        impl $crate::types::DerefToPyAny for $name {}
157
158        impl $name {
159            fn type_object_raw(py: $crate::Python<'_>) -> *mut $crate::ffi::PyTypeObject {
160                use $crate::types::PyTypeMethods;
161                static TYPE_OBJECT: $crate::impl_::exceptions::ImportedExceptionTypeObject =
162                    $crate::impl_::exceptions::ImportedExceptionTypeObject::new(
163                        stringify!($module),
164                        stringify!($name),
165                    );
166                TYPE_OBJECT.get(py).as_type_ptr()
167            }
168        }
169    };
170}

172/// Defines a new exception type.
173///
174/// # Syntax
175///
176/// * `module` is the name of the containing module.
177/// * `name` is the name of the new exception type.
178/// * `base` is the base class of `MyError`, usually [`PyException`].
179/// * `doc` (optional) is the docstring visible to users (with `.__doc__` and `help()`) and
180/// accompanies your error type in your crate's documentation.
181///
182/// # Examples
183///
184/// ```
185/// use pyo3::prelude::*;
186/// use pyo3::create_exception;
187/// use pyo3::exceptions::PyException;
188///
189/// create_exception!(my_module, MyError, PyException, "Some description.");
190///
191/// #[pyfunction]
192/// fn raise_myerror() -> PyResult<()> {
193///     let err = MyError::new_err("Some error happened.");
194///     Err(err)
195/// }
196///
197/// #[pymodule]
198/// fn my_module(m: &Bound<'_, PyModule>) -> PyResult<()> {
199///     m.add("MyError", m.py().get_type_bound::<MyError>())?;
200///     m.add_function(wrap_pyfunction!(raise_myerror, m)?)?;
201///     Ok(())
202/// }
203/// # fn main() -> PyResult<()> {
204/// #     Python::with_gil(|py| -> PyResult<()> {
205/// #         let fun = wrap_pyfunction_bound!(raise_myerror, py)?;
206/// #         let locals = pyo3::types::PyDict::new_bound(py);
207/// #         locals.set_item("MyError", py.get_type_bound::<MyError>())?;
208/// #         locals.set_item("raise_myerror", fun)?;
209/// #
210/// #         py.run_bound(
211/// # "try:
212/// #     raise_myerror()
213/// # except MyError as e:
214/// #     assert e.__doc__ == 'Some description.'
215/// #     assert str(e) == 'Some error happened.'",
216/// #             None,
217/// #             Some(&locals),
218/// #         )?;
219/// #
220/// #         Ok(())
221/// #     })
222/// # }
223/// ```
224///
225/// Python code can handle this exception like any other exception:
226///
227/// ```python
228/// from my_module import MyError, raise_myerror
229///
230/// try:
231///     raise_myerror()
232/// except MyError as e:
233///     assert e.__doc__ == 'Some description.'
234///     assert str(e) == 'Some error happened.'
235/// ```
236///
237#[macro_export]
238macro_rules! create_exception {
239    ($module: expr, $name: ident, $base: ty) => {
240        #[repr(transparent)]
241        #[allow(non_camel_case_types)] // E.g. `socket.herror`
242        pub struct $name($crate::PyAny);
243
244        $crate::impl_exception_boilerplate!($name);
245
246        $crate::create_exception_type_object!($module, $name, $base, ::std::option::Option::None);
247    };
248    ($module: expr, $name: ident, $base: ty, $doc: expr) => {
249        #[repr(transparent)]
250        #[allow(non_camel_case_types)] // E.g. `socket.herror`
251        #[doc = $doc]
252        pub struct $name($crate::PyAny);
253
254        $crate::impl_exception_boilerplate!($name);
255
256        $crate::create_exception_type_object!(
257            $module,
258            $name,
259            $base,
260            ::std::option::Option::Some($doc)
261        );
262    };
263}

265/// `impl PyTypeInfo for $name` where `$name` is an
266/// exception newly defined in Rust code.
267#[doc(hidden)]
268#[macro_export]
269macro_rules! create_exception_type_object {
270    ($module: expr, $name: ident, $base: ty, $doc: expr) => {
271        $crate::pyobject_native_type_core!(
272            $name,
273            $name::type_object_raw,
274            #module=::std::option::Option::Some(stringify!($module))
275        );
276
277        impl $name {
278            fn type_object_raw(py: $crate::Python<'_>) -> *mut $crate::ffi::PyTypeObject {
279                use $crate::sync::GILOnceCell;
280                static TYPE_OBJECT: GILOnceCell<$crate::Py<$crate::types::PyType>> =
281                    GILOnceCell::new();
282
283                TYPE_OBJECT
284                    .get_or_init(py, ||
285                        $crate::PyErr::new_type_bound(
286                            py,
287                            concat!(stringify!($module), ".", stringify!($name)),
288                            $doc,
289                            ::std::option::Option::Some(&py.get_type_bound::<$base>()),
290                            ::std::option::Option::None,
291                        ).expect("Failed to initialize new exception type.")
292                ).as_ptr() as *mut $crate::ffi::PyTypeObject
293            }
294        }
295    };
296}

651impl PyUnicodeDecodeError {
652    /// Deprecated form of [`PyUnicodeDecodeError::new_bound`].
653    #[cfg_attr(
654        not(feature = "gil-refs"),
655        deprecated(
656            since = "0.21.0",
657            note = "`PyUnicodeDecodeError::new` will be replaced by `PyUnicodeDecodeError::new_bound` in a future PyO3 version"
658        )
659    )]
660    pub fn new<'p>(
661        py: Python<'p>,
662        encoding: &CStr,
663        input: &[u8],
664        range: ops::Range<usize>,
665        reason: &CStr,
666    ) -> PyResult<&'p PyUnicodeDecodeError> {
667        Ok(PyUnicodeDecodeError::new_bound(py, encoding, input, range, reason)?.into_gil_ref())
668    }
669
670    /// Creates a Python `UnicodeDecodeError`.
671    pub fn new_bound<'p>(
672        py: Python<'p>,
673        encoding: &CStr,
674        input: &[u8],
675        range: ops::Range<usize>,
676        reason: &CStr,
677    ) -> PyResult<Bound<'p, PyUnicodeDecodeError>> {
678        use crate::ffi_ptr_ext::FfiPtrExt;
679        use crate::py_result_ext::PyResultExt;
680        unsafe {
681            ffi::PyUnicodeDecodeError_Create(
682                encoding.as_ptr(),
683                input.as_ptr() as *const c_char,
684                input.len() as ffi::Py_ssize_t,
685                range.start as ffi::Py_ssize_t,
686                range.end as ffi::Py_ssize_t,
687                reason.as_ptr(),
688            )
689            .assume_owned_or_err(py)
690        }
691        .downcast_into()
692    }

694    /// Deprecated form of [`PyUnicodeDecodeError::new_utf8_bound`].
695    #[cfg_attr(
696        not(feature = "gil-refs"),
697        deprecated(
698            since = "0.21.0",
699            note = "`PyUnicodeDecodeError::new_utf8` will be replaced by `PyUnicodeDecodeError::new_utf8_bound` in a future PyO3 version"
700        )
701    )]
702    pub fn new_utf8<'p>(
703        py: Python<'p>,
704        input: &[u8],
705        err: std::str::Utf8Error,
706    ) -> PyResult<&'p PyUnicodeDecodeError> {
707        Ok(PyUnicodeDecodeError::new_utf8_bound(py, input, err)?.into_gil_ref())
708    }

710    /// Creates a Python `UnicodeDecodeError` from a Rust UTF-8 decoding error.
711    ///
712    /// # Examples
713    ///
714    /// ```
715    /// #![cfg_attr(invalid_from_utf8_lint, allow(invalid_from_utf8))]
716    /// use pyo3::prelude::*;
717    /// use pyo3::exceptions::PyUnicodeDecodeError;
718    ///
719    /// # fn main() -> PyResult<()> {
720    /// Python::with_gil(|py| {
721    ///     let invalid_utf8 = b"fo\xd8o";
722    ///     let err = std::str::from_utf8(invalid_utf8).expect_err("should be invalid utf8");
723    ///     let decode_err = PyUnicodeDecodeError::new_utf8_bound(py, invalid_utf8, err)?;
724    ///     assert_eq!(
725    ///         decode_err.to_string(),
726    ///         "'utf-8' codec can't decode byte 0xd8 in position 2: invalid utf-8"
727    ///     );
728    ///     Ok(())
729    /// })
730    /// # }
731    pub fn new_utf8_bound<'p>(
732        py: Python<'p>,
733        input: &[u8],
734        err: std::str::Utf8Error,
735    ) -> PyResult<Bound<'p, PyUnicodeDecodeError>> {
736        let pos = err.valid_up_to();
737        PyUnicodeDecodeError::new_bound(
738            py,
739            CStr::from_bytes_with_nul(b"utf-8\0").unwrap(),
740            input,
741            pos..(pos + 1),
742            CStr::from_bytes_with_nul(b"invalid utf-8\0").unwrap(),
743        )
744    }
745}

829            })
830        }
831    };
