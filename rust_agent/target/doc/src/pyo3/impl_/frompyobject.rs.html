5pub enum Extractor<'a, 'py, T> {
6    Bound(fn(&'a Bound<'py, PyAny>) -> PyResult<T>),
7    GilRef(fn(&'a PyAny) -> PyResult<T>),
8}

10impl<'a, 'py, T> From<fn(&'a Bound<'py, PyAny>) -> PyResult<T>> for Extractor<'a, 'py, T> {
11    fn from(value: fn(&'a Bound<'py, PyAny>) -> PyResult<T>) -> Self {
12        Self::Bound(value)
13    }
14}

16impl<'a, T> From<fn(&'a PyAny) -> PyResult<T>> for Extractor<'a, '_, T> {
17    fn from(value: fn(&'a PyAny) -> PyResult<T>) -> Self {
18        Self::GilRef(value)
19    }

22impl<'a, 'py, T> Extractor<'a, 'py, T> {
23    pub(crate) fn call(self, obj: &'a Bound<'py, PyAny>) -> PyResult<T> {
24        match self {
25            Extractor::Bound(f) => f(obj),
26            Extractor::GilRef(f) => f(obj.as_gil_ref()),
27        }
28    }
29}

31#[cold]
32pub fn failed_to_extract_enum(
33    py: Python<'_>,
34    type_name: &str,
35    variant_names: &[&str],
36    error_names: &[&str],
37    errors: &[PyErr],
38) -> PyErr {
39    // TODO maybe use ExceptionGroup on Python 3.11+ ?
40    let mut err_msg = format!(
41        "failed to extract enum {} ('{}')",
42        type_name,
43        error_names.join(" | ")
44    );
45    for ((variant_name, error_name), error) in variant_names.iter().zip(error_names).zip(errors) {
46        use std::fmt::Write;
47        write!(
48            &mut err_msg,
49            "\n- variant {variant_name} ({error_name}): {error_msg}",
50            variant_name = variant_name,
51            error_name = error_name,
52            error_msg = extract_traceback(py, error.clone_ref(py)),
53        )
54        .unwrap();
55    }
56    PyTypeError::new_err(err_msg)
57}
58
59/// Flattens a chain of errors into a single string.
60fn extract_traceback(py: Python<'_>, mut error: PyErr) -> String {
61    use std::fmt::Write;
62
63    let mut error_msg = error.to_string();
64    while let Some(cause) = error.cause(py) {
65        write!(&mut error_msg, ", caused by {}", cause).unwrap();
66        error = cause
67    }
68    error_msg
69}

71pub fn extract_struct_field<'py, T>(
72    obj: &Bound<'py, PyAny>,
73    struct_name: &str,
74    field_name: &str,
75) -> PyResult<T>
76where
77    T: FromPyObject<'py>,
78{
79    match obj.extract() {
80        Ok(value) => Ok(value),
81        Err(err) => Err(failed_to_extract_struct_field(
82            obj.py(),
83            err,
84            struct_name,
85            field_name,
86        )),
87    }

108fn failed_to_extract_struct_field(
109    py: Python<'_>,
110    inner_err: PyErr,
111    struct_name: &str,
112    field_name: &str,
113) -> PyErr {
114    let new_err = PyTypeError::new_err(format!(
115        "failed to extract field {}.{}",
116        struct_name, field_name
117    ));
118    new_err.set_cause(py, ::std::option::Option::Some(inner_err));
119    new_err
