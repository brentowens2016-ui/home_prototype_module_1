8/// Trait to enable the use of `wrap_pyfunction` with both `Python` and `PyModule`,
9/// and also to infer the return type of either `&'py PyCFunction` or `Bound<'py, PyCFunction>`.
10pub trait WrapPyFunctionArg<'py, T> {
11    fn wrap_pyfunction(self, method_def: &PyMethodDef) -> PyResult<T>;
12}
13
14impl<'py> WrapPyFunctionArg<'py, Bound<'py, PyCFunction>> for Bound<'py, PyModule> {
15    fn wrap_pyfunction(self, method_def: &PyMethodDef) -> PyResult<Bound<'py, PyCFunction>> {
16        PyCFunction::internal_new(self.py(), method_def, Some(&self))
17    }
18}
19
20impl<'py> WrapPyFunctionArg<'py, Bound<'py, PyCFunction>> for &'_ Bound<'py, PyModule> {
21    fn wrap_pyfunction(self, method_def: &PyMethodDef) -> PyResult<Bound<'py, PyCFunction>> {
22        PyCFunction::internal_new(self.py(), method_def, Some(self))
23    }
24}

38// For Python<'py>, only the GIL Ref form exists to avoid causing type inference to kick in.
39// The `wrap_pyfunction_bound!` macro is needed for the Bound form.
40impl<'py> WrapPyFunctionArg<'py, &'py PyCFunction> for Python<'py> {
41    fn wrap_pyfunction(self, method_def: &PyMethodDef) -> PyResult<&'py PyCFunction> {
42        PyCFunction::internal_new(self, method_def, None).map(Bound::into_gil_ref)
43    }
44}

46impl<'py> WrapPyFunctionArg<'py, &'py PyCFunction> for &'py PyModule {
47    fn wrap_pyfunction(self, method_def: &PyMethodDef) -> PyResult<&'py PyCFunction> {
48        PyCFunction::internal_new(self.py(), method_def, Some(&self.as_borrowed()))
49            .map(Bound::into_gil_ref)
50    }
51}

53/// Helper for `wrap_pyfunction_bound!` to guarantee return type of `Bound<'py, PyCFunction>`.
54pub struct OnlyBound<T>(pub T);
55
56impl<'py, T> WrapPyFunctionArg<'py, Bound<'py, PyCFunction>> for OnlyBound<T>
57where
58    T: WrapPyFunctionArg<'py, Bound<'py, PyCFunction>>,
59{
60    fn wrap_pyfunction(self, method_def: &PyMethodDef) -> PyResult<Bound<'py, PyCFunction>> {

62    }
63}
