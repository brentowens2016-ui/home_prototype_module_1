21/// `Sync` wrapper of `ffi::PyModuleDef`.
22pub struct ModuleDef {
23    // wrapped in UnsafeCell so that Rust compiler treats this as interior mutability
24    ffi_def: UnsafeCell<ffi::PyModuleDef>,
25    initializer: ModuleInitializer,
26    /// Interpreter ID where module was initialized (not applicable on PyPy).
27    #[cfg(all(
28        not(any(PyPy, GraalPy)),
29        Py_3_9,
30        not(all(windows, Py_LIMITED_API, not(Py_3_10)))
31    ))]
32    interpreter: AtomicI64,
33    /// Initialized module object, cached to avoid reinitialization.
34    module: GILOnceCell<Py<PyModule>>,
35}
36
37/// Wrapper to enable initializer to be used in const fns.
38pub struct ModuleInitializer(pub for<'py> fn(&Bound<'py, PyModule>) -> PyResult<()>);
39
40unsafe impl Sync for ModuleDef {}

42impl ModuleDef {
43    /// Make new module definition with given module name.
44    ///
45    /// # Safety
46    /// `name` and `doc` must be null-terminated strings.
47    pub const unsafe fn new(
48        name: &'static str,
49        doc: &'static str,
50        initializer: ModuleInitializer,
51    ) -> Self {
52        const INIT: ffi::PyModuleDef = ffi::PyModuleDef {
53            m_base: ffi::PyModuleDef_HEAD_INIT,
54            m_name: std::ptr::null(),
55            m_doc: std::ptr::null(),
56            m_size: 0,
57            m_methods: std::ptr::null_mut(),
58            m_slots: std::ptr::null_mut(),
59            m_traverse: None,
60            m_clear: None,
61            m_free: None,
62        };
63
64        let ffi_def = UnsafeCell::new(ffi::PyModuleDef {
65            m_name: name.as_ptr() as *const _,
66            m_doc: doc.as_ptr() as *const _,
67            ..INIT
68        });
69
70        ModuleDef {
71            ffi_def,
72            initializer,
73            // -1 is never expected to be a valid interpreter ID
74            #[cfg(all(
75                not(any(PyPy, GraalPy)),
76                Py_3_9,
77                not(all(windows, Py_LIMITED_API, not(Py_3_10)))
78            ))]
79            interpreter: AtomicI64::new(-1),
80            module: GILOnceCell::new(),
81        }
82    }
83    /// Builds a module using user given initializer. Used for [`#[pymodule]`][crate::pymodule].
84    pub fn make_module(&'static self, py: Python<'_>) -> PyResult<Py<PyModule>> {
85        #[cfg(all(PyPy, not(Py_3_8)))]
86        {
87            use crate::types::any::PyAnyMethods;
88            const PYPY_GOOD_VERSION: [u8; 3] = [7, 3, 8];
89            let version = py
90                .import_bound("sys")?
91                .getattr("implementation")?
92                .getattr("version")?;
93            if version.lt(crate::types::PyTuple::new_bound(py, PYPY_GOOD_VERSION))? {
94                let warn = py.import_bound("warnings")?.getattr("warn")?;
95                warn.call1((
96                    "PyPy 3.7 versions older than 7.3.8 are known to have binary \
97                        compatibility issues which may cause segfaults. Please upgrade.",
98                ))?;
99            }
100        }
101        // Check the interpreter ID has not changed, since we currently have no way to guarantee
102        // that static data is not reused across interpreters.
103        //
104        // PyPy does not have subinterpreters, so no need to check interpreter ID.
105        #[cfg(not(any(PyPy, GraalPy)))]
106        {
107            // PyInterpreterState_Get is only available on 3.9 and later, but is missing
108            // from python3.dll for Windows stable API on 3.9
109            #[cfg(all(Py_3_9, not(all(windows, Py_LIMITED_API, not(Py_3_10)))))]
110            {
111                let current_interpreter =
112                    unsafe { ffi::PyInterpreterState_GetID(ffi::PyInterpreterState_Get()) };
113                crate::err::error_on_minusone(py, current_interpreter)?;
114                if let Err(initialized_interpreter) = self.interpreter.compare_exchange(
115                    -1,
116                    current_interpreter,
117                    Ordering::SeqCst,
118                    Ordering::SeqCst,
119                ) {
120                    if initialized_interpreter != current_interpreter {
121                        return Err(PyImportError::new_err(
122                            "PyO3 modules do not yet support subinterpreters, see https://github.com/PyO3/pyo3/issues/576",
123                        ));
124                    }
125                }
126            }
127            #[cfg(not(all(Py_3_9, not(all(windows, Py_LIMITED_API, not(Py_3_10))))))]
128            {
129                // CPython before 3.9 does not have APIs to check the interpreter ID, so best that can be
130                // done to guard against subinterpreters is fail if the module is initialized twice
131                if self.module.get(py).is_some() {
132                    return Err(PyImportError::new_err(
133                        "PyO3 modules compiled for CPython 3.8 or older may only be initialized once per interpreter process"
134                    ));
135                }
136            }
137        }
138        self.module
139            .get_or_try_init(py, || {
140                let module = unsafe {
141                    Py::<PyModule>::from_owned_ptr_or_err(
142                        py,
143                        ffi::PyModule_Create(self.ffi_def.get()),
144                    )?
145                };
146                self.initializer.0(module.bind(py))?;
147                Ok(module)
148            })
149            .map(|py_module| py_module.clone_ref(py))
150    }
151}

153/// Trait to add an element (class, function...) to a module.
154///
155/// Currently only implemented for classes.
156pub trait PyAddToModule {
157    fn add_to_module(&'static self, module: &Bound<'_, PyModule>) -> PyResult<()>;
158}
159
160/// For adding native types (non-pyclass) to a module.
161pub struct AddTypeToModule<T>(PhantomData<T>);
162
163impl<T> AddTypeToModule<T> {
164    #[allow(clippy::new_without_default)]
165    pub const fn new() -> Self {
166        AddTypeToModule(PhantomData)
167    }
168}
169
170impl<T: PyTypeInfo> PyAddToModule for AddTypeToModule<T> {
171    fn add_to_module(&'static self, module: &Bound<'_, PyModule>) -> PyResult<()> {
172        module.add(T::NAME, T::type_object_bound(module.py()))
173    }
174}
175
176/// For adding a class to a module.
177pub struct AddClassToModule<T>(PhantomData<T>);
178
179impl<T> AddClassToModule<T> {
180    #[allow(clippy::new_without_default)]
181    pub const fn new() -> Self {
182        AddClassToModule(PhantomData)
183    }
184}
185
186impl<T: PyClass> PyAddToModule for AddClassToModule<T> {
187    fn add_to_module(&'static self, module: &Bound<'_, PyModule>) -> PyResult<()> {
188        module.add_class::<T>()
189    }
190}
191
192/// For adding a function to a module.
193impl PyAddToModule for PyMethodDef {
194    fn add_to_module(&'static self, module: &Bound<'_, PyModule>) -> PyResult<()> {
195        module.add_function(PyCFunction::internal_new(module.py(), self, Some(module))?)
196    }
197}
198
199/// For adding a module to a module.
200impl PyAddToModule for ModuleDef {
201    fn add_to_module(&'static self, module: &Bound<'_, PyModule>) -> PyResult<()> {
202        module.add_submodule(self.make_module(module.py())?.bind(module.py()))
203    }
204}

207mod tests {
208    use std::{
209        borrow::Cow,
210        sync::atomic::{AtomicBool, Ordering},
211    };

213    use crate::{

218    use super::{ModuleDef, ModuleInitializer};

220    #[test]
221    fn module_init() {
222        static MODULE_DEF: ModuleDef = unsafe {
223            ModuleDef::new(
224                "test_module\0",
225                "some doc\0",
226                ModuleInitializer(|m| {
227                    m.add("SOME_CONSTANT", 42)?;
228                    Ok(())
229                }),
230            )
231        };
232        Python::with_gil(|py| {
233            let module = MODULE_DEF.make_module(py).unwrap().into_bound(py);
234            assert_eq!(
235                module
236                    .getattr("__name__")
237                    .unwrap()
238                    .extract::<Cow<'_, str>>()
239                    .unwrap(),
240                "test_module",
241            );
242            assert_eq!(
243                module
244                    .getattr("__doc__")
245                    .unwrap()
246                    .extract::<Cow<'_, str>>()
247                    .unwrap(),
248                "some doc",
249            );
250            assert_eq!(
251                module
252                    .getattr("SOME_CONSTANT")
253                    .unwrap()
254                    .extract::<u8>()
255                    .unwrap(),
256                42,
257            );
258        })
259    }

261    #[test]
262    fn module_def_new() {
263        // To get coverage for ModuleDef::new() need to create a non-static ModuleDef, however init
264        // etc require static ModuleDef, so this test needs to be separated out.
265        static NAME: &str = "test_module\0";
266        static DOC: &str = "some doc\0";
267
268        static INIT_CALLED: AtomicBool = AtomicBool::new(false);
269
270        #[allow(clippy::unnecessary_wraps)]
271        fn init(_: &Bound<'_, PyModule>) -> PyResult<()> {
272            INIT_CALLED.store(true, Ordering::SeqCst);
273            Ok(())
274        }
275
276        unsafe {
277            let module_def: ModuleDef = ModuleDef::new(NAME, DOC, ModuleInitializer(init));
278            assert_eq!((*module_def.ffi_def.get()).m_name, NAME.as_ptr() as _);
279            assert_eq!((*module_def.ffi_def.get()).m_doc, DOC.as_ptr() as _);
280
281            Python::with_gil(|py| {
282                module_def.initializer.0(&py.import_bound("builtins").unwrap()).unwrap();
283                assert!(INIT_CALLED.load(Ordering::SeqCst));
284            })
285        }
286    }
