1//! Trampolines for various pyfunction and pymethod implementations.
2//!
3//! They exist to monomorphise std::panic::catch_unwind once into PyO3, rather than inline in every
4//! function, thus saving a huge amount of compile-time complexity.

19#[inline]
20pub unsafe fn module_init(
21    f: for<'py> unsafe fn(Python<'py>) -> PyResult<Py<PyModule>>,
22) -> *mut ffi::PyObject {
23    trampoline(|py| f(py).map(|module| module.into_ptr()))

26#[inline]
27#[allow(clippy::used_underscore_binding)]
28pub unsafe fn noargs(
29    slf: *mut ffi::PyObject,
30    _args: *mut ffi::PyObject,
31    f: for<'py> unsafe fn(Python<'py>, *mut ffi::PyObject) -> PyResult<*mut ffi::PyObject>,
32) -> *mut ffi::PyObject {
33    #[cfg(not(GraalPy))] // this is not specified and GraalPy does not pass null here
34    debug_assert!(_args.is_null());
35    trampoline(|py| f(py, slf))
36}

38macro_rules! trampoline {
39    (pub fn $name:ident($($arg_names:ident: $arg_types:ty),* $(,)?) -> $ret:ty;) => {
40        #[inline]
41        pub unsafe fn $name(
42            $($arg_names: $arg_types,)*
43            f: for<'py> unsafe fn (Python<'py>, $($arg_types),*) -> PyResult<$ret>,
44        ) -> $ret {
45            trampoline(|py| f(py, $($arg_names,)*))
46        }
47    }
48}
49
50macro_rules! trampolines {
51    ($(pub fn $name:ident($($arg_names:ident: $arg_types:ty),* $(,)?) -> $ret:ty);* ;) => {
52        $(trampoline!(pub fn $name($($arg_names: $arg_types),*) -> $ret;));*;
53    }
54}

141#[inline]
142pub(crate) unsafe fn dealloc(
143    slf: *mut ffi::PyObject,
144    f: for<'py> unsafe fn(Python<'py>, *mut ffi::PyObject) -> (),
145) {
146    // After calling tp_dealloc the object is no longer valid,
147    // so pass null_mut() to the context.
148    //
149    // (Note that we don't allow the implementation `f` to fail.)
150    trampoline_unraisable(
151        |py| {
152            f(py, slf);
153            Ok(())
154        },
155        std::ptr::null_mut(),
156    )
157}

170/// Implementation of trampoline functions, which sets up a GILPool and calls F.
171///
172/// Panics during execution are trapped so that they don't propagate through any
173/// outer FFI boundary.
174#[inline]
175pub(crate) fn trampoline<F, R>(body: F) -> R
176where
177    F: for<'py> FnOnce(Python<'py>) -> PyResult<R> + UnwindSafe,
178    R: PyCallbackOutput,
179{
180    let trap = PanicTrap::new("uncaught panic at ffi boundary");
181    // Necessary to construct a pool until PyO3 0.22 when the GIL Refs API is fully disabled
182    #[allow(deprecated)]
183    let pool = unsafe { GILPool::new() };
184    let py = pool.python();
185    let out = panic_result_into_callback_output(
186        py,
187        panic::catch_unwind(move || -> PyResult<_> { body(py) }),
188    );
189    trap.disarm();
190    out
191}

193/// Converts the output of std::panic::catch_unwind into a Python function output, either by raising a Python
194/// exception or by unwrapping the contained success output.
195#[inline]
196fn panic_result_into_callback_output<R>(
197    py: Python<'_>,
198    panic_result: Result<PyResult<R>, Box<dyn Any + Send + 'static>>,
199) -> R
200where
201    R: PyCallbackOutput,
202{
203    let py_err = match panic_result {
204        Ok(Ok(value)) => return value,
205        Ok(Err(py_err)) => py_err,
206        Err(payload) => PanicException::from_panic_payload(payload),
207    };
208    py_err.restore(py);
209    R::ERR_VALUE
210}

212/// Implementation of trampoline for functions which can't return an error.
213///
214/// Panics during execution are trapped so that they don't propagate through any
215/// outer FFI boundary.
216///
217/// Exceptions produced are sent to `sys.unraisablehook`.
218///
219/// # Safety
220///
221/// ctx must be either a valid ffi::PyObject or NULL
222#[inline]
223unsafe fn trampoline_unraisable<F>(body: F, ctx: *mut ffi::PyObject)
224where
225    F: for<'py> FnOnce(Python<'py>) -> PyResult<()> + UnwindSafe,
226{
227    let trap = PanicTrap::new("uncaught panic at ffi boundary");
228    // Necessary to construct a pool until PyO3 0.22 when the GIL Refs API is fully disabled
229    #[allow(deprecated)]
230    let pool = GILPool::new();
231    let py = pool.python();
232    if let Err(py_err) = panic::catch_unwind(move || body(py))
233        .unwrap_or_else(|payload| Err(PanicException::from_panic_payload(payload)))
234    {
235        py_err.write_unraisable_bound(py, ctx.assume_borrowed_or_opt(py).as_deref());
236    }
237    trap.disarm();
238}
