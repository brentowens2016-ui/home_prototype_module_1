18/// Types that are built into the Python interpreter.
19///
20/// PyO3 is designed in a way that all references to those types are bound
21/// to the GIL, which is why you can get a token from all references of those
22/// types.
23///
24/// # Safety
25///
26/// This trait must only be implemented for types which cannot be accessed without the GIL.
27pub unsafe trait PyNativeType: Sized {
28    /// The form of this which is stored inside a `Py<T>` smart pointer.
29    type AsRefSource: HasPyGilRef<AsRefTarget = Self>;
30
31    /// Cast `&self` to a `Borrowed` smart pointer.
32    ///
33    /// `Borrowed<T>` implements `Deref<Target=Bound<T>>`, so can also be used in locations
34    /// where `Bound<T>` is expected.
35    ///
36    /// This is available as a migration tool to adjust code from the deprecated "GIL Refs"
37    /// API to the `Bound` smart pointer API.
38    #[inline]
39    fn as_borrowed(&self) -> Borrowed<'_, '_, Self::AsRefSource> {
40        // Safety: &'py Self is expected to be a Python pointer,
41        // so has the same layout as Borrowed<'py, 'py, T>
42        Borrowed(
43            unsafe { NonNull::new_unchecked(self as *const Self as *mut _) },
44            PhantomData,
45            self.py(),
46        )
47    }
48
49    /// Returns a GIL marker constrained to the lifetime of this type.
50    #[inline]
51    fn py(&self) -> Python<'_> {
52        unsafe { Python::assume_gil_acquired() }
53    }
54    /// Cast `&PyAny` to `&Self` without no type checking.
55    ///
56    /// # Safety
57    ///
58    /// `obj` must have the same layout as `*const ffi::PyObject` and must be
59    /// an instance of a type corresponding to `Self`.
60    unsafe fn unchecked_downcast(obj: &PyAny) -> &Self {
61        &*(obj.as_ptr() as *const Self)
62    }
63}

715/// A GIL-independent reference to an object allocated on the Python heap.
716///
717/// This type does not auto-dereference to the inner object because you must prove you hold the GIL to access it.
718/// Instead, call one of its methods to access the inner object:
719///  - [`Py::as_ref`], to borrow a GIL-bound reference to the contained object.
720///  - [`Py::borrow`], [`Py::try_borrow`], [`Py::borrow_mut`], or [`Py::try_borrow_mut`],
721/// to get a (mutable) reference to a contained pyclass, using a scheme similar to std's [`RefCell`].
722/// See the [`PyCell` guide entry](https://pyo3.rs/latest/class.html#pycell-and-interior-mutability)
723/// for more information.
724///  - You can call methods directly on `Py` with [`Py::call`], [`Py::call_method`] and friends.
725/// These require passing in the [`Python<'py>`](crate::Python) token but are otherwise similar to the corresponding
726/// methods on [`PyAny`].
727///
728/// # Example: Storing Python objects in `#[pyclass]` structs
729///
730/// Usually `Bound<'py, T>` is recommended for interacting with Python objects as its lifetime `'py`
731/// is an association to the GIL and that enables many operations to be done as efficiently as possible.
732///
733/// However, `#[pyclass]` structs cannot carry a lifetime, so `Py<T>` is the only way to store
734/// a Python object in a `#[pyclass]` struct.
735///
736/// For example, this won't compile:
737///
738/// ```compile_fail
739/// # use pyo3::prelude::*;
740/// # use pyo3::types::PyDict;
741/// #
742/// #[pyclass]
743/// struct Foo<'py> {
744///     inner: Bound<'py, PyDict>,
745/// }
746///
747/// impl Foo {
748///     fn new() -> Foo {
749///         let foo = Python::with_gil(|py| {
750///             // `py` will only last for this scope.
751///
752///             // `Bound<'py, PyDict>` inherits the GIL lifetime from `py` and
753///             // so won't be able to outlive this closure.
754///             let dict: Bound<'_, PyDict> = PyDict::new_bound(py);
755///
756///             // because `Foo` contains `dict` its lifetime
757///             // is now also tied to `py`.
758///             Foo { inner: dict }
759///         });
760///         // Foo is no longer valid.
761///         // Returning it from this function is a  compiler error 
762///         foo
763///     }
764/// }
765/// ```
766///
767/// [`Py`]`<T>` can be used to get around this by converting `dict` into a GIL-independent reference:
768///
769/// ```rust
770/// use pyo3::prelude::*;
771/// use pyo3::types::PyDict;
772///
773/// #[pyclass]
774/// struct Foo {
775///     inner: Py<PyDict>,
776/// }
777///
778/// #[pymethods]
779/// impl Foo {
780///     #[new]
781///     fn __new__() -> Foo {
782///         Python::with_gil(|py| {
783///             let dict: Py<PyDict> = PyDict::new_bound(py).unbind();
784///             Foo { inner: dict }
785///         })
786///     }
787/// }
788/// #
789/// # fn main() -> PyResult<()> {
790/// #     Python::with_gil(|py| {
791/// #         let m = pyo3::types::PyModule::new_bound(py, "test")?;
792/// #         m.add_class::<Foo>()?;
793/// #
794/// #         let foo: Bound<'_, Foo> = m.getattr("Foo")?.call0()?.downcast_into()?;
795/// #         let dict = &foo.borrow().inner;
796/// #         let dict: &Bound<'_, PyDict> = dict.bind(py);
797/// #
798/// #         Ok(())
799/// #     })
800/// # }
801/// ```
802///
803/// This can also be done with other pyclasses:
804/// ```rust
805/// use pyo3::prelude::*;
806///
807/// #[pyclass]
808/// struct Bar {/* ... */}
809///
810/// #[pyclass]
811/// struct Foo {
812///     inner: Py<Bar>,
813/// }
814///
815/// #[pymethods]
816/// impl Foo {
817///     #[new]
818///     fn __new__() -> PyResult<Foo> {
819///         Python::with_gil(|py| {
820///             let bar: Py<Bar> = Py::new(py, Bar {})?;
821///             Ok(Foo { inner: bar })
822///         })
823///     }
824/// }
825/// #
826/// # fn main() -> PyResult<()> {
827/// #     Python::with_gil(|py| {
828/// #         let m = pyo3::types::PyModule::new_bound(py, "test")?;
829/// #         m.add_class::<Foo>()?;
830/// #
831/// #         let foo: Bound<'_, Foo> = m.getattr("Foo")?.call0()?.downcast_into()?;
832/// #         let bar = &foo.borrow().inner;
833/// #         let bar: &Bar = &*bar.borrow(py);
834/// #
835/// #         Ok(())
836/// #     })
837/// # }
838/// ```
839///
840/// # Example: Shared ownership of Python objects
841///
842/// `Py<T>` can be used to share ownership of a Python object, similar to std's [`Rc`]`<T>`.
843/// As with [`Rc`]`<T>`, cloning it increases its reference count rather than duplicating
844/// the underlying object.
845///
846/// This can be done using either [`Py::clone_ref`] or [`Py`]`<T>`'s [`Clone`] trait implementation.
847/// [`Py::clone_ref`] will be faster if you happen to be already holding the GIL.
848///
849/// ```rust
850/// use pyo3::prelude::*;
851/// use pyo3::types::PyDict;
852///
853/// # fn main() {
854/// Python::with_gil(|py| {
855///     let first: Py<PyDict> = PyDict::new_bound(py).unbind();
856///
857///     // All of these are valid syntax
858///     let second = Py::clone_ref(&first, py);
859///     let third = first.clone_ref(py);
860///     let fourth = Py::clone(&first);
861///     let fifth = first.clone();
862///
863///     // Disposing of our original `Py<PyDict>` just decrements the reference count.
864///     drop(first);
865///
866///     // They all point to the same object
867///     assert!(second.is(&third));
868///     assert!(fourth.is(&fifth));
869///     assert!(second.is(&fourth));
870/// });
871/// # }
872/// ```
873///
874/// # Preventing reference cycles
875///
876/// It is easy to accidentally create reference cycles using [`Py`]`<T>`.
877/// The Python interpreter can break these reference cycles within pyclasses if they
878/// [integrate with the garbage collector][gc]. If your pyclass contains other Python
879/// objects you should implement it to avoid leaking memory.
880///
881/// # A note on Python reference counts
882///
883/// Dropping a [`Py`]`<T>` will eventually decrease Python's reference count
884/// of the pointed-to variable, allowing Python's garbage collector to free
885/// the associated memory, but this may not happen immediately.  This is
886/// because a [`Py`]`<T>` can be dropped at any time, but the Python reference
887/// count can only be modified when the GIL is held.
888///
889/// If a [`Py`]`<T>` is dropped while its thread happens to be holding the
890/// GIL then the Python reference count will be decreased immediately.
891/// Otherwise, the reference count will be decreased the next time the GIL is
892/// reacquired.
893///
894/// If you happen to be already holding the GIL, [`Py::drop_ref`] will decrease
895/// the Python reference count immediately and will execute slightly faster than
896/// relying on implicit [`Drop`]s.
897///
898/// # A note on `Send` and `Sync`
899///
900/// Accessing this object is threadsafe, since any access to its API requires a [`Python<'py>`](crate::Python) token.
901/// As you can only get this by acquiring the GIL, `Py<...>` implements [`Send`] and [`Sync`].
902///
903/// [`Rc`]: std::rc::Rc
904/// [`RefCell`]: std::cell::RefCell
905/// [gc]: https://pyo3.rs/main/class/protocols.html#garbage-collector-integration
906#[repr(transparent)]
907pub struct Py<T>(NonNull<ffi::PyObject>, PhantomData<T>);
908
909// The inner value is only accessed through ways that require proving the gil is held
910#[cfg(feature = "nightly")]
911unsafe impl<T> crate::marker::Ungil for Py<T> {}
912unsafe impl<T> Send for Py<T> {}
913unsafe impl<T> Sync for Py<T> {}

947    ///
948    /// By binding to the GIL lifetime, this allows the GIL-bound reference to not require
949    /// [`Python<'py>`](crate::Python) for any of its methods, which makes calling methods
950    /// on it more ergonomic.
951    ///
952    /// For native types, this reference is `&T`. For pyclasses, this is `&PyCell<T>`.
953    ///
954    /// Note that the lifetime of the returned reference is the shortest of `&self` and
955    /// [`Python<'py>`](crate::Python).
956    /// Consider using [`Py::into_ref`] instead if this poses a problem.
957    ///
958    /// # Examples
959    ///
960    /// Get access to `&PyList` from `Py<PyList>`:
961    ///
962    /// ```

968    ///     # #[allow(deprecated)]

974    /// Get access to `&PyCell<MyClass>` from `Py<MyClass>`:
975    ///
976    /// ```
977    /// # use pyo3::prelude::*;
978    /// #
979    /// #[pyclass]
980    /// struct MyClass {}
981    ///
982    /// Python::with_gil(|py| {
983    ///     let my_class: Py<MyClass> = Py::new(py, MyClass {}).unwrap();
984    ///     # #[allow(deprecated)]
985    ///     let my_class_cell: &PyCell<MyClass> = my_class.as_ref(py);
986    ///     assert!(my_class_cell.try_borrow().is_ok());
987    /// });
988    /// ```

990        not(feature = "gil-refs"),
991        deprecated(
992            since = "0.21.0",
993            note = "use `obj.bind(py)` instead of `obj.as_ref(py)`"
994        )
995    )]

997        let any = self.as_ptr() as *const PyAny;
998        unsafe { PyNativeType::unchecked_downcast(&*any) }
999    }

1001    /// Borrows a GIL-bound reference to the contained `T` independently of the lifetime of `T`.
1002    ///
1003    /// This method is similar to [`as_ref`](#method.as_ref) but consumes `self` and registers the
1004    /// Python object reference in PyO3's object storage. The reference count for the Python
1005    /// object will not be decreased until the GIL lifetime ends.
1006    ///
1007    /// You should prefer using [`as_ref`](#method.as_ref) if you can as it'll have less overhead.
1008    ///
1009    /// # Examples
1010    ///
1011    /// [`Py::as_ref`]'s lifetime limitation forbids creating a function that references a
1012    /// variable created inside the function.
1013    ///
1014    /// ```rust,compile_fail
1015    /// # use pyo3::prelude::*;
1016    /// #
1017    /// fn new_py_any<'py>(py: Python<'py>, value: impl IntoPy<Py<PyAny>>) -> &'py PyAny {
1018    ///     let obj: Py<PyAny> = value.into_py(py);
1019    ///
1020    ///     // The lifetime of the return value of this function is the shortest
1021    ///     // of `obj` and `py`. As `obj` is owned by the current function,
1022    ///     // Rust won't let the return value escape this function!
1023    ///     obj.as_ref(py)
1024    /// }
1025    /// ```
1026    ///
1027    /// This can be solved by using [`Py::into_ref`] instead, which does not suffer from this issue.
1028    /// Note that the lifetime of the [`Python<'py>`](crate::Python) token is transferred to
1029    /// the returned reference.
1030    ///
1031    /// ```rust
1032    /// # use pyo3::prelude::*;
1033    /// # #[allow(dead_code)] // This is just to show it compiles.
1034    /// fn new_py_any<'py>(py: Python<'py>, value: impl IntoPy<Py<PyAny>>) -> &'py PyAny {
1035    ///     let obj: Py<PyAny> = value.into_py(py);
1036    ///
1037    ///     // This reference's lifetime is determined by `py`'s lifetime.
1038    ///     // Because that originates from outside this function,
1039    ///     // this return value is allowed.
1040    ///     # #[allow(deprecated)]
1041    ///     obj.into_ref(py)
1042    /// }
1043    /// ```
1044    #[cfg_attr(
1045        not(feature = "gil-refs"),
1046        deprecated(
1047            since = "0.21.0",
1048            note = "use `obj.into_bound(py)` instead of `obj.into_ref(py)`"
1049        )
1050    )]
1051    pub fn into_ref(self, py: Python<'_>) -> &T::AsRefTarget {
1052        #[allow(deprecated)]
1053        unsafe {
1054            py.from_owned_ptr(self.into_ptr())
1055        }
1056    }
1057}

1872/// A commonly-used alias for `Py<PyAny>`.
1873///
1874/// This is an owned reference a Python object without any type information. This value can also be
1875/// safely sent between threads.
1876///
1877/// See the documentation for [`Py`](struct.Py.html).
1878pub type PyObject = Py<PyAny>;
1879
1880impl PyObject {
1881    /// Deprecated form of [`PyObject::downcast_bound`]
1882    #[cfg_attr(
1883        not(feature = "gil-refs"),
1884        deprecated(
1885            since = "0.21.0",
1886            note = "`PyObject::downcast` will be replaced by `PyObject::downcast_bound` in a future PyO3 version"
1887        )
1888    )]
1889    #[inline]
1890    pub fn downcast<'py, T>(&'py self, py: Python<'py>) -> Result<&'py T, PyDowncastError<'py>>
1891    where
1892        T: PyTypeCheck<AsRefTarget = T>,
1893    {
1894        self.downcast_bound::<T>(py)
1895            .map(Bound::as_gil_ref)
1896            .map_err(PyDowncastError::from_downcast_err)
1897    }
1898    /// Downcast this `PyObject` to a concrete Python type or pyclass.
1899    ///
1900    /// Note that you can often avoid downcasting yourself by just specifying
1901    /// the desired type in function or method signatures.
1902    /// However, manual downcasting is sometimes necessary.
1903    ///
1904    /// For extracting a Rust-only type, see [`Py::extract`](struct.Py.html#method.extract).
1905    ///
1906    /// # Example: Downcasting to a specific Python object
1907    ///
1908    /// ```rust
1909    /// use pyo3::prelude::*;
1910    /// use pyo3::types::{PyDict, PyList};
1911    ///
1912    /// Python::with_gil(|py| {
1913    ///     let any: PyObject = PyDict::new_bound(py).into();
1914    ///
1915    ///     assert!(any.downcast_bound::<PyDict>(py).is_ok());
1916    ///     assert!(any.downcast_bound::<PyList>(py).is_err());
1917    /// });
1918    /// ```
1919    ///
1920    /// # Example: Getting a reference to a pyclass
1921    ///
1922    /// This is useful if you want to mutate a `PyObject` that
1923    /// might actually be a pyclass.
1924    ///
1925    /// ```rust
1926    /// # fn main() -> Result<(), pyo3::PyErr> {
1927    /// use pyo3::prelude::*;
1928    ///
1929    /// #[pyclass]
1930    /// struct Class {
1931    ///     i: i32,
1932    /// }
1933    ///
1934    /// Python::with_gil(|py| {
1935    ///     let class: PyObject = Py::new(py, Class { i: 0 }).unwrap().into_py(py);
1936    ///
1937    ///     let class_bound = class.downcast_bound::<Class>(py)?;
1938    ///
1939    ///     class_bound.borrow_mut().i += 1;
1940    ///
1941    ///     // Alternatively you can get a `PyRefMut` directly
1942    ///     let class_ref: PyRefMut<'_, Class> = class.extract(py)?;
1943    ///     assert_eq!(class_ref.i, 1);
1944    ///     Ok(())
1945    /// })
1946    /// # }
1947    /// ```
1948    #[inline]
1949    pub fn downcast_bound<'py, T>(
1950        &self,
1951        py: Python<'py>,
1952    ) -> Result<&Bound<'py, T>, DowncastError<'_, 'py>>
1953    where
1954        T: PyTypeCheck,
1955    {
1956        self.bind(py).downcast()
1957    }
1958
1959    /// Deprecated form of [`PyObject::downcast_bound_unchecked`]
1960    ///
1961    /// # Safety
1962    ///
1963    /// Callers must ensure that the type is valid or risk type confusion.
1964    #[cfg_attr(
1965        not(feature = "gil-refs"),
1966        deprecated(
1967            since = "0.21.0",
1968            note = "`PyObject::downcast_unchecked` will be replaced by `PyObject::downcast_bound_unchecked` in a future PyO3 version"
1969        )
1970    )]
1971    #[inline]
1972    pub unsafe fn downcast_unchecked<'py, T>(&'py self, py: Python<'py>) -> &T
1973    where
1974        T: HasPyGilRef<AsRefTarget = T>,
1975    {
1976        self.downcast_bound_unchecked::<T>(py).as_gil_ref()
1977    }
1978
1979    /// Casts the PyObject to a concrete Python object type without checking validity.
1980    ///
1981    /// # Safety
1982    ///
1983    /// Callers must ensure that the type is valid or risk type confusion.
1984    #[inline]
1985    pub unsafe fn downcast_bound_unchecked<'py, T>(&self, py: Python<'py>) -> &Bound<'py, T> {
1986        self.bind(py).downcast_unchecked()
1987    }
1988}
