13/// Value with concurrent access protected by the GIL.
14///
15/// This is a synchronization primitive based on Python's global interpreter lock (GIL).
16/// It ensures that only one thread at a time can access the inner value via shared references.
17/// It can be combined with interior mutability to obtain mutable references.
18///
19/// # Example
20///
21/// Combining `GILProtected` with `RefCell` enables mutable access to static data:
22///
23/// ```
24/// # use pyo3::prelude::*;
25/// use pyo3::sync::GILProtected;
26/// use std::cell::RefCell;
27///
28/// static NUMBERS: GILProtected<RefCell<Vec<i32>>> = GILProtected::new(RefCell::new(Vec::new()));
29///
30/// Python::with_gil(|py| {
31///     NUMBERS.get(py).borrow_mut().push(42);
32/// });
33/// ```
34pub struct GILProtected<T> {
35    value: T,
36}
37
38impl<T> GILProtected<T> {
39    /// Place the given value under the protection of the GIL.
40    pub const fn new(value: T) -> Self {
41        Self { value }
42    }
43
44    /// Gain access to the inner value by giving proof of having acquired the GIL.
45    pub fn get<'py>(&'py self, _py: Python<'py>) -> &'py T {
46        &self.value
47    }
48
49    /// Gain access to the inner value by giving proof that garbage collection is happening.
50    pub fn traverse<'py>(&'py self, _visit: PyVisit<'py>) -> &'py T {
51        &self.value
52    }
53}
54
55unsafe impl<T> Sync for GILProtected<T> where T: Send {}

57/// A write-once cell similar to [`once_cell::OnceCell`](https://docs.rs/once_cell/latest/once_cell/).
58///
59/// Unlike `once_cell::sync` which blocks threads to achieve thread safety, this implementation
60/// uses the Python GIL to mediate concurrent access. This helps in cases where `once_cell` or
61/// `lazy_static`'s synchronization strategy can lead to deadlocks when interacting with the Python
62/// GIL. For an example, see [the FAQ section](https://pyo3.rs/latest/faq.html) of the guide.
63///
64/// Note that:
65///  1) `get_or_init` and `get_or_try_init` do not protect against infinite recursion
66///     from reentrant initialization.
67///  2) If the initialization function `f` provided to `get_or_init` (or `get_or_try_init`)
68///     temporarily releases the GIL (e.g. by calling `Python::import`) then it is possible
69///     for a second thread to also begin initializing the `GITOnceCell`. Even when this
70///     happens `GILOnceCell` guarantees that only **one** write to the cell ever occurs
71///     - this is treated as a race, other threads will discard the value they compute and
72///     return the result of the first complete computation.
73///
74/// # Examples
75///
76/// The following example shows how to use `GILOnceCell` to share a reference to a Python list
77/// between threads:
78///
79/// ```
80/// use pyo3::sync::GILOnceCell;
81/// use pyo3::prelude::*;
82/// use pyo3::types::PyList;
83///
84/// static LIST_CELL: GILOnceCell<Py<PyList>> = GILOnceCell::new();
85///
86/// pub fn get_shared_list(py: Python<'_>) -> &Bound<'_, PyList> {
87///     LIST_CELL
88///         .get_or_init(py, || PyList::empty_bound(py).unbind())
89///         .bind(py)
90/// }
91/// # Python::with_gil(|py| assert_eq!(get_shared_list(py).len(), 0));
92/// ```
93#[derive(Default)]
94pub struct GILOnceCell<T>(UnsafeCell<Option<T>>);
95
96// T: Send is needed for Sync because the thread which drops the GILOnceCell can be different
97// to the thread which fills it.
98unsafe impl<T: Send + Sync> Sync for GILOnceCell<T> {}
99unsafe impl<T: Send> Send for GILOnceCell<T> {}
100
101impl<T> GILOnceCell<T> {
102    /// Create a `GILOnceCell` which does not yet contain a value.
103    pub const fn new() -> Self {
104        Self(UnsafeCell::new(None))
105    }
106
107    /// Get a reference to the contained value, or `None` if the cell has not yet been written.
108    #[inline]
109    pub fn get(&self, _py: Python<'_>) -> Option<&T> {
110        // Safe because if the cell has not yet been written, None is returned.
111        unsafe { &*self.0.get() }.as_ref()
112    }
113
114    /// Get a reference to the contained value, initializing it if needed using the provided
115    /// closure.
116    ///
117    /// See the type-level documentation for detail on re-entrancy and concurrent initialization.
118    #[inline]
119    pub fn get_or_init<F>(&self, py: Python<'_>, f: F) -> &T
120    where
121        F: FnOnce() -> T,
122    {
123        if let Some(value) = self.get(py) {
124            return value;
125        }
126
127        match self.init(py, || Ok::<T, std::convert::Infallible>(f())) {
128            Ok(value) => value,
129            Err(void) => match void {},
130        }
131    }
132
133    /// Like `get_or_init`, but accepts a fallible initialization function. If it fails, the cell
134    /// is left uninitialized.
135    ///
136    /// See the type-level documentation for detail on re-entrancy and concurrent initialization.
137    #[inline]
138    pub fn get_or_try_init<F, E>(&self, py: Python<'_>, f: F) -> Result<&T, E>
139    where
140        F: FnOnce() -> Result<T, E>,
141    {
142        if let Some(value) = self.get(py) {
143            return Ok(value);
144        }
145
146        self.init(py, f)
147    }
148
149    #[cold]
150    fn init<F, E>(&self, py: Python<'_>, f: F) -> Result<&T, E>
151    where
152        F: FnOnce() -> Result<T, E>,
153    {
154        // Note that f() could temporarily release the GIL, so it's possible that another thread
155        // writes to this GILOnceCell before f() finishes. That's fine; we'll just have to discard
156        // the value computed here and accept a bit of wasted computation.
157        let value = f()?;
158        let _ = self.set(py, value);
159
160        Ok(self.get(py).unwrap())
161    }
162
163    /// Get the contents of the cell mutably. This is only possible if the reference to the cell is
164    /// unique.
165    pub fn get_mut(&mut self) -> Option<&mut T> {
166        self.0.get_mut().as_mut()
167    }
168
169    /// Set the value in the cell.
170    ///
171    /// If the cell has already been written, `Err(value)` will be returned containing the new
172    /// value which was not written.
173    pub fn set(&self, _py: Python<'_>, value: T) -> Result<(), T> {
174        // Safe because GIL is held, so no other thread can be writing to this cell concurrently.
175        let inner = unsafe { &mut *self.0.get() };
176        if inner.is_some() {
177            return Err(value);
178        }
179
180        *inner = Some(value);
181        Ok(())
182    }
183
184    /// Takes the value out of the cell, moving it back to an uninitialized state.
185    ///
186    /// Has no effect and returns None if the cell has not yet been written.
187    pub fn take(&mut self) -> Option<T> {
188        self.0.get_mut().take()
189    }
190
191    /// Consumes the cell, returning the wrapped value.
192    ///
193    /// Returns None if the cell has not yet been written.
194    pub fn into_inner(self) -> Option<T> {
195        self.0.into_inner()
196    }
197}

199impl GILOnceCell<Py<PyType>> {
200    /// Get a reference to the contained Python type, initializing it if needed.
201    ///
202    /// This is a shorthand method for `get_or_init` which imports the type from Python on init.
203    pub(crate) fn get_or_try_init_type_ref<'py>(
204        &self,
205        py: Python<'py>,
206        module_name: &str,
207        attr_name: &str,
208    ) -> PyResult<&Bound<'py, PyType>> {
209        self.get_or_try_init(py, || {
210            let type_object = py
211                .import_bound(module_name)?
212                .getattr(attr_name)?
213                .downcast_into()?;
214            Ok(type_object.unbind())
215        })
216        .map(|ty| ty.bind(py))
217    }
218}

220/// Interns `text` as a Python string and stores a reference to it in static storage.
221///
222/// A reference to the same Python string is returned on each invocation.
223///
224/// # Example: Using `intern!` to avoid needlessly recreating the same Python string
225///
226/// ```
227/// use pyo3::intern;
228/// # use pyo3::{prelude::*, types::PyDict};
229///
230/// #[pyfunction]
231/// fn create_dict(py: Python<'_>) -> PyResult<Bound<'_, PyDict>> {
232///     let dict = PyDict::new_bound(py);
233///     //             ðŸ‘‡ A new `PyString` is created
234///     //                for every call of this function.
235///     dict.set_item("foo", 42)?;
236///     Ok(dict)
237/// }
238///
239/// #[pyfunction]
240/// fn create_dict_faster(py: Python<'_>) -> PyResult<Bound<'_, PyDict>> {
241///     let dict = PyDict::new_bound(py);
242///     //               ðŸ‘‡ A `PyString` is created once and reused
243///     //                  for the lifetime of the program.
244///     dict.set_item(intern!(py, "foo"), 42)?;
245///     Ok(dict)
246/// }
247/// #
248/// # Python::with_gil(|py| {
249/// #     let fun_slow = wrap_pyfunction_bound!(create_dict, py).unwrap();
250/// #     let dict = fun_slow.call0().unwrap();
251/// #     assert!(dict.contains("foo").unwrap());
252/// #     let fun = wrap_pyfunction_bound!(create_dict_faster, py).unwrap();
253/// #     let dict = fun.call0().unwrap();
254/// #     assert!(dict.contains("foo").unwrap());
255/// # });
256/// ```
257#[macro_export]
258macro_rules! intern {
259    ($py: expr, $text: expr) => {{
260        static INTERNED: $crate::sync::Interned = $crate::sync::Interned::new($text);
261        INTERNED.get($py)
262    }};
263}
264
265/// Implementation detail for `intern!` macro.
266#[doc(hidden)]
267pub struct Interned(&'static str, GILOnceCell<Py<PyString>>);
268
269impl Interned {
270    /// Creates an empty holder for an interned `str`.
271    pub const fn new(value: &'static str) -> Self {
272        Interned(value, GILOnceCell::new())
273    }
274
275    /// Gets or creates the interned `str` value.
276    #[inline]
277    pub fn get<'py>(&self, py: Python<'py>) -> &Bound<'py, PyString> {
278        self.1
279            .get_or_init(py, || PyString::intern_bound(py, self.0).into())
280            .bind(py)
281    }
282}

290    #[test]
291    fn test_intern() {
292        Python::with_gil(|py| {
293            let foo1 = "foo";
294            let foo2 = intern!(py, "foo");
295            let foo3 = intern!(py, stringify!(foo));
296
297            let dict = PyDict::new_bound(py);
298            dict.set_item(foo1, 42_usize).unwrap();
299            assert!(dict.contains(foo2).unwrap());
300            assert_eq!(
301                dict.get_item(foo3)
302                    .unwrap()
303                    .unwrap()
304                    .extract::<usize>()
305                    .unwrap(),
306                42
307            );
308        });
309    }
310
311    #[test]
312    fn test_once_cell() {
313        Python::with_gil(|py| {
314            let mut cell = GILOnceCell::new();
315
316            assert!(cell.get(py).is_none());
317
318            assert_eq!(cell.get_or_try_init(py, || Err(5)), Err(5));
319            assert!(cell.get(py).is_none());
320
321            assert_eq!(cell.get_or_try_init(py, || Ok::<_, ()>(2)), Ok(&2));
322            assert_eq!(cell.get(py), Some(&2));
323
324            assert_eq!(cell.get_or_try_init(py, || Err(5)), Ok(&2));
325
326            assert_eq!(cell.take(), Some(2));
327            assert_eq!(cell.into_inner(), None)
328        })
329    }
