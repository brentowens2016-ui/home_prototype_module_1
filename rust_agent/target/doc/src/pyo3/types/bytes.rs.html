31    /// Creates a new Python bytestring object.
32    /// The bytestring is initialized by copying the data from the `&[u8]`.
33    ///
34    /// Panics if out of memory.
35    pub fn new_bound<'p>(py: Python<'p>, s: &[u8]) -> Bound<'p, PyBytes> {
36        let ptr = s.as_ptr() as *const c_char;
37        let len = s.len() as ffi::Py_ssize_t;
38        unsafe {
39            ffi::PyBytes_FromStringAndSize(ptr, len)
40                .assume_owned(py)
41                .downcast_into_unchecked()
42        }
43    }

60    /// Creates a new Python `bytes` object with an `init` closure to write its contents.
61    /// Before calling `init` the bytes' contents are zero-initialised.
62    /// * If Python raises a MemoryError on the allocation, `new_with` will return
63    ///   it inside `Err`.
64    /// * If `init` returns `Err(e)`, `new_with` will return `Err(e)`.
65    /// * If `init` returns `Ok(())`, `new_with` will return `Ok(&PyBytes)`.
66    ///
67    /// # Examples
68    ///
69    /// ```
70    /// use pyo3::{prelude::*, types::PyBytes};
71    ///
72    /// # fn main() -> PyResult<()> {
73    /// Python::with_gil(|py| -> PyResult<()> {
74    ///     let py_bytes = PyBytes::new_bound_with(py, 10, |bytes: &mut [u8]| {
75    ///         bytes.copy_from_slice(b"Hello Rust");
76    ///         Ok(())
77    ///     })?;
78    ///     let bytes: &[u8] = py_bytes.extract()?;
79    ///     assert_eq!(bytes, b"Hello Rust");
80    ///     Ok(())
81    /// })
82    /// # }
83    /// ```
84    pub fn new_bound_with<F>(py: Python<'_>, len: usize, init: F) -> PyResult<Bound<'_, PyBytes>>
85    where
86        F: FnOnce(&mut [u8]) -> PyResult<()>,
87    {
88        unsafe {
89            let pyptr = ffi::PyBytes_FromStringAndSize(std::ptr::null(), len as ffi::Py_ssize_t);
90            // Check for an allocation error and return it
91            let pybytes = pyptr.assume_owned_or_err(py)?.downcast_into_unchecked();
92            let buffer: *mut u8 = ffi::PyBytes_AsString(pyptr).cast();
93            debug_assert!(!buffer.is_null());
94            // Zero-initialise the uninitialised bytestring
95            std::ptr::write_bytes(buffer, 0u8, len);
96            // (Further) Initialise the bytestring in init
97            // If init returns an Err, pypybytearray will automatically deallocate the buffer
98            init(std::slice::from_raw_parts_mut(buffer, len)).map(|_| pybytes)
99        }
100    }

117    /// Creates a new Python byte string object from a raw pointer and length.
118    ///
119    /// Panics if out of memory.
120    ///
121    /// # Safety
122    ///
123    /// This function dereferences the raw pointer `ptr` as the
124    /// leading pointer of a slice of length `len`. [As with
125    /// `std::slice::from_raw_parts`, this is
126    /// unsafe](https://doc.rust-lang.org/std/slice/fn.from_raw_parts.html#safety).
127    pub unsafe fn bound_from_ptr(py: Python<'_>, ptr: *const u8, len: usize) -> Bound<'_, PyBytes> {
128        ffi::PyBytes_FromStringAndSize(ptr as *const _, len as isize)
129            .assume_owned(py)
130            .downcast_into_unchecked()
131    }

158impl<'a> Borrowed<'a, '_, PyBytes> {
159    /// Gets the Python string as a byte slice.
160    #[allow(clippy::wrong_self_convention)]
161    pub(crate) fn as_bytes(self) -> &'a [u8] {
162        unsafe {
163            let buffer = ffi::PyBytes_AsString(self.as_ptr()) as *const u8;
164            let length = ffi::PyBytes_Size(self.as_ptr()) as usize;
165            debug_assert!(!buffer.is_null());
166            std::slice::from_raw_parts(buffer, length)
167        }
168    }
169}
