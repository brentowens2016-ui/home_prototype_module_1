34fn ensure_datetime_api(py: Python<'_>) -> PyResult<&'static PyDateTime_CAPI> {
35    if let Some(api) = unsafe { pyo3_ffi::PyDateTimeAPI().as_ref() } {
36        Ok(api)
37    } else {
38        unsafe {
39            PyDateTime_IMPORT();
40            pyo3_ffi::PyDateTimeAPI().as_ref()
41        }
42        .ok_or_else(|| PyErr::fetch(py))
43    }
44}
45
46fn expect_datetime_api(py: Python<'_>) -> &'static PyDateTime_CAPI {
47    ensure_datetime_api(py).expect("failed to import `datetime` C API")
48}

50// Type Check macros
51//
52// These are bindings around the C API typecheck macros, all of them return
53// `1` if True and `0` if False. In all type check macros, the argument (`op`)
54// must not be `NULL`. The implementations here all call ensure_datetime_api
55// to ensure that the PyDateTimeAPI is initialized before use
56//
57//
58// # Safety
59//
60// These functions must only be called when the GIL is held!
61
62macro_rules! ffi_fun_with_autoinit {
63    ($(#[$outer:meta] unsafe fn $name: ident($arg: ident: *mut PyObject) -> $ret: ty;)*) => {
64        $(
65            #[$outer]
66            #[allow(non_snake_case)]
67            /// # Safety
68            ///
69            /// Must only be called while the GIL is held
70            unsafe fn $name($arg: *mut crate::ffi::PyObject) -> $ret {
71
72                let _ = ensure_datetime_api(Python::assume_gil_acquired());
73                crate::ffi::$name($arg)
74            }
75        )*
76
77
78    };
79}

100/// Trait for accessing the date components of a struct containing a date.
101pub trait PyDateAccess {
102    /// Returns the year, as a positive int.
103    ///
104    /// Implementations should conform to the upstream documentation:
105    /// <https://docs.python.org/3/c-api/datetime.html#c.PyDateTime_GET_YEAR>
106    fn get_year(&self) -> i32;
107    /// Returns the month, as an int from 1 through 12.
108    ///
109    /// Implementations should conform to the upstream documentation:
110    /// <https://docs.python.org/3/c-api/datetime.html#c.PyDateTime_GET_MONTH>
111    fn get_month(&self) -> u8;
112    /// Returns the day, as an int from 1 through 31.
113    ///
114    /// Implementations should conform to the upstream documentation:
115    /// <https://docs.python.org/3/c-api/datetime.html#c.PyDateTime_GET_DAY>
116    fn get_day(&self) -> u8;
117}

119/// Trait for accessing the components of a struct containing a timedelta.
120///
121/// Note: These access the individual components of a (day, second,
122/// microsecond) representation of the delta, they are *not* intended as
123/// aliases for calculating the total duration in each of these units.
124pub trait PyDeltaAccess {
125    /// Returns the number of days, as an int from -999999999 to 999999999.

129    fn get_days(&self) -> i32;
130    /// Returns the number of seconds, as an int from 0 through 86399.
131    ///
132    /// Implementations should conform to the upstream documentation:
133    /// <https://docs.python.org/3/c-api/datetime.html#c.PyDateTime_DELTA_GET_DAYS>
134    fn get_seconds(&self) -> i32;
135    /// Returns the number of microseconds, as an int from 0 through 999999.
136    ///
137    /// Implementations should conform to the upstream documentation:
138    /// <https://docs.python.org/3/c-api/datetime.html#c.PyDateTime_DELTA_GET_DAYS>
139    fn get_microseconds(&self) -> i32;
140}

554impl<'py> PyTzInfoAccess<'py> for Bound<'py, PyDateTime> {
555    fn get_tzinfo_bound(&self) -> Option<Bound<'py, PyTzInfo>> {
556        let ptr = self.as_ptr() as *mut ffi::PyDateTime_DateTime;
557        #[cfg(not(GraalPy))]
558        unsafe {
559            if (*ptr).hastzinfo != 0 {
560                Some(
561                    (*ptr)
562                        .tzinfo
563                        .assume_borrowed(self.py())
564                        .to_owned()
565                        .downcast_into_unchecked(),
566                )
567            } else {
568                None
569            }
570        }
571
572        #[cfg(GraalPy)]
573        unsafe {
574            let res = PyDateTime_DATE_GET_TZINFO(ptr as *mut ffi::PyObject);
575            if Py_IsNone(res) == 1 {
576                None
577            } else {
578                Some(
579                    res.assume_borrowed(self.py())
580                        .to_owned()
581                        .downcast_into_unchecked(),
582                )
583            }
584        }
585    }

662        minute: u8,
663        second: u8,
664        microsecond: u32,
665        tzinfo: Option<&'py PyTzInfo>,
666        fold: bool,
667    ) -> PyResult<&'py PyTime> {
668        Self::new_bound_with_fold(
669            py,
670            hour,
671            minute,
672            second,
673            microsecond,
674            tzinfo.map(PyTzInfo::as_borrowed).as_deref(),
675            fold,
676        )
677        .map(Bound::into_gil_ref)
678    }
679
680    /// Alternate constructor that takes a `fold` argument. See [`PyDateTime::new_with_fold`].
681    pub fn new_bound_with_fold<'py>(
682        py: Python<'py>,
683        hour: u8,
684        minute: u8,
685        second: u8,
686        microsecond: u32,
687        tzinfo: Option<&Bound<'py, PyTzInfo>>,
688        fold: bool,
689    ) -> PyResult<Bound<'py, PyTime>> {
690        let api = ensure_datetime_api(py)?;
691        unsafe {
692            (api.Time_FromTimeAndFold)(
693                c_int::from(hour),
694                c_int::from(minute),
695                c_int::from(second),
696                microsecond as c_int,
697                opt_to_pyobj(tzinfo),
698                fold as c_int,
699                api.TimeType,
700            )
701            .assume_owned_or_err(py)
702            .downcast_into_unchecked()
703        }
704    }
705}

757impl<'py> PyTzInfoAccess<'py> for Bound<'py, PyTime> {
758    fn get_tzinfo_bound(&self) -> Option<Bound<'py, PyTzInfo>> {
759        let ptr = self.as_ptr() as *mut ffi::PyDateTime_Time;
760        #[cfg(not(GraalPy))]
761        unsafe {
762            if (*ptr).hastzinfo != 0 {
763                Some(
764                    (*ptr)
765                        .tzinfo
766                        .assume_borrowed(self.py())
767                        .to_owned()
768                        .downcast_into_unchecked(),
769                )
770            } else {
771                None
772            }
773        }
774
775        #[cfg(GraalPy)]
776        unsafe {
777            let res = PyDateTime_TIME_GET_TZINFO(ptr as *mut ffi::PyObject);
778            if Py_IsNone(res) == 1 {
779                None
780            } else {
781                Some(
782                    res.assume_borrowed(self.py())
783                        .to_owned()
784                        .downcast_into_unchecked(),
785                )
786            }
787        }
788    }

818/// Equivalent to `datetime.timezone.utc`
819pub fn timezone_utc_bound(py: Python<'_>) -> Bound<'_, PyTzInfo> {

823    unsafe {

825            .TimeZone_UTC
826            .assume_borrowed(py)

828            .downcast_into_unchecked()
