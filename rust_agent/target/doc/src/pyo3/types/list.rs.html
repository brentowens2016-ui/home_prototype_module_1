20#[inline]
21#[track_caller]
22pub(crate) fn new_from_iter<'py>(
23    py: Python<'py>,
24    elements: &mut dyn ExactSizeIterator<Item = PyObject>,
25) -> Bound<'py, PyList> {
26    unsafe {
27        // PyList_New checks for overflow but has a bad error message, so we check ourselves
28        let len: Py_ssize_t = elements
29            .len()
30            .try_into()
31            .expect("out of range integral type conversion attempted on `elements.len()`");
32
33        let ptr = ffi::PyList_New(len);
34
35        // We create the `Bound` pointer here for two reasons:
36        // - panics if the ptr is null
37        // - its Drop cleans up the list if user code or the asserts panic.
38        let list = ptr.assume_owned(py).downcast_into_unchecked();
39
40        let mut counter: Py_ssize_t = 0;
41
42        for obj in elements.take(len as usize) {
43            #[cfg(not(Py_LIMITED_API))]
44            ffi::PyList_SET_ITEM(ptr, counter, obj.into_ptr());
45            #[cfg(Py_LIMITED_API)]
46            ffi::PyList_SetItem(ptr, counter, obj.into_ptr());
47            counter += 1;
48        }
49
50        assert!(elements.next().is_none(), "Attempted to create PyList but `elements` was larger than reported by its `ExactSizeIterator` implementation.");
51        assert_eq!(len, counter, "Attempted to create PyList but `elements` was smaller than reported by its `ExactSizeIterator` implementation.");
52
53        list
54    }
55}

76    /// Constructs a new list with the given elements.
77    ///
78    /// If you want to create a [`PyList`] with elements of different or unknown types, or from an
79    /// iterable that doesn't implement [`ExactSizeIterator`], use [`PyList::append`].
80    ///
81    /// # Examples
82    ///
83    /// ```rust
84    /// use pyo3::prelude::*;
85    /// use pyo3::types::PyList;
86    ///
87    /// # fn main() {
88    /// Python::with_gil(|py| {
89    ///     let elements: Vec<i32> = vec![0, 1, 2, 3, 4, 5];

92    /// });
93    /// # }

95    ///
96    /// # Panics
97    ///
98    /// This function will panic if `element`'s [`ExactSizeIterator`] implementation is incorrect.

100    /// function with (for example) [`Vec`]`<T>` or `&[T]` will always succeed.

634pub struct BoundListIterator<'py> {
635    list: Bound<'py, PyList>,
636    index: usize,
637    length: usize,
638}
639
640impl<'py> BoundListIterator<'py> {
641    fn new(list: Bound<'py, PyList>) -> Self {
642        let length: usize = list.len();
643        BoundListIterator {
644            list,
645            index: 0,
646            length,
647        }
648    }
649
650    unsafe fn get_item(&self, index: usize) -> Bound<'py, PyAny> {
651        #[cfg(any(Py_LIMITED_API, PyPy))]
652        let item = self.list.get_item(index).expect("list.get failed");
653        #[cfg(not(any(Py_LIMITED_API, PyPy)))]
654        let item = self.list.get_item_unchecked(index);
655        item
656    }
657}
658
659impl<'py> Iterator for BoundListIterator<'py> {
660    type Item = Bound<'py, PyAny>;
661
662    #[inline]
663    fn next(&mut self) -> Option<Self::Item> {
664        let length = self.length.min(self.list.len());
665
666        if self.index < length {
667            let item = unsafe { self.get_item(self.index) };
668            self.index += 1;
669            Some(item)
670        } else {
671            None
672        }
673    }
674
675    #[inline]
676    fn size_hint(&self) -> (usize, Option<usize>) {
677        let len = self.len();
678        (len, Some(len))
679    }
680}
681
682impl DoubleEndedIterator for BoundListIterator<'_> {
683    #[inline]
684    fn next_back(&mut self) -> Option<Self::Item> {
685        let length = self.length.min(self.list.len());
686
687        if self.index < length {
688            let item = unsafe { self.get_item(length - 1) };
689            self.length = length - 1;
690            Some(item)
691        } else {
692            None
693        }
694    }
695}
696
697impl ExactSizeIterator for BoundListIterator<'_> {
698    fn len(&self) -> usize {
699        self.length.saturating_sub(self.index)
700    }
701}
702
703impl FusedIterator for BoundListIterator<'_> {}
704
705impl<'py> IntoIterator for Bound<'py, PyList> {
706    type Item = Bound<'py, PyAny>;
707    type IntoIter = BoundListIterator<'py>;
708
709    fn into_iter(self) -> Self::IntoIter {
710        BoundListIterator::new(self)
711    }
712}

714impl<'py> IntoIterator for &Bound<'py, PyList> {
715    type Item = Bound<'py, PyAny>;
716    type IntoIter = BoundListIterator<'py>;
717
718    fn into_iter(self) -> Self::IntoIter {
719        self.iter()
720    }
721}

1263    #[cfg(feature = "macros")]
1264    #[test]
1265    fn bad_clone_mem_leaks() {
1266        use crate::{Py, PyAny};
1267        use std::sync::atomic::{AtomicUsize, Ordering::SeqCst};
1268        static NEEDS_DESTRUCTING_COUNT: AtomicUsize = AtomicUsize::new(0);
1269
1270        #[crate::pyclass]
1271        #[pyo3(crate = "crate")]
1272        struct Bad(usize);
1273
1274        impl Clone for Bad {
1275            fn clone(&self) -> Self {
1276                // This panic should not lead to a memory leak
1277                assert_ne!(self.0, 42);
1278                NEEDS_DESTRUCTING_COUNT.fetch_add(1, SeqCst);
1279
1280                Bad(self.0)
1281            }
1282        }
1283
1284        impl Drop for Bad {
1285            fn drop(&mut self) {
1286                NEEDS_DESTRUCTING_COUNT.fetch_sub(1, SeqCst);
1287            }
1288        }
1289
1290        impl ToPyObject for Bad {
1291            fn to_object(&self, py: Python<'_>) -> Py<PyAny> {
1292                self.to_owned().into_py(py)
1293            }
1294        }
1295
1296        struct FaultyIter(Range<usize>, usize);
1297
1298        impl Iterator for FaultyIter {
1299            type Item = Bad;
1300
1301            fn next(&mut self) -> Option<Self::Item> {
1302                self.0.next().map(|i| {
1303                    NEEDS_DESTRUCTING_COUNT.fetch_add(1, SeqCst);
1304                    Bad(i)
1305                })
1306            }
1307        }
1308
1309        impl ExactSizeIterator for FaultyIter {
1310            fn len(&self) -> usize {
1311                self.1
1312            }
1313        }
1314
1315        Python::with_gil(|py| {
1316            std::panic::catch_unwind(|| {
1317                let iter = FaultyIter(0..50, 50);
1318                let _list = PyList::new(py, iter);
1319            })
1320            .unwrap_err();
1321        });
1322
1323        assert_eq!(
1324            NEEDS_DESTRUCTING_COUNT.load(SeqCst),
1325            0,
1326            "Some destructors did not run"
1327        );
1328    }
