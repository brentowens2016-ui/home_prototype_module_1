30/// Adds all the [`#[cfg]` flags](index.html) to the current compilation.
31///
32/// This should be called from a build script.
33///
34/// The full list of attributes added are the following:
35///
36/// | Flag | Description |
37/// | ---- | ----------- |
38/// | `#[cfg(Py_3_7)]`, `#[cfg(Py_3_8)]`, `#[cfg(Py_3_9)]`, `#[cfg(Py_3_10)]` | These attributes mark code only for a given Python version and up. For example, `#[cfg(Py_3_7)]` marks code which can run on Python 3.7 **and newer**. |
39/// | `#[cfg(Py_LIMITED_API)]` | This marks code which is run when compiling with PyO3's `abi3` feature enabled. |
40/// | `#[cfg(PyPy)]` | This marks code which is run when compiling for PyPy. |
41/// | `#[cfg(GraalPy)]` | This marks code which is run when compiling for GraalPy. |
42///
43/// For examples of how to use these attributes, [see PyO3's guide](https://pyo3.rs/latest/building-and-distribution/multiple_python_versions.html).
44#[cfg(feature = "resolve-config")]
45pub fn use_pyo3_cfgs() {
46    for cargo_command in get().build_script_outputs() {
47        println!("{}", cargo_command)
48    }
49}

51/// Adds linker arguments suitable for PyO3's `extension-module` feature.
52///
53/// This should be called from a build script.
54///
55/// The following link flags are added:
56/// - macOS: `-undefined dynamic_lookup`
57/// - wasm32-unknown-emscripten: `-sSIDE_MODULE=2 -sWASM_BIGINT`
58///
59/// All other platforms currently are no-ops, however this may change as necessary
60/// in future.
61pub fn add_extension_module_link_args() {
62    _add_extension_module_link_args(&impl_::target_triple_from_env(), std::io::stdout())
63}
64
65fn _add_extension_module_link_args(triple: &Triple, mut writer: impl std::io::Write) {
66    if triple.operating_system == OperatingSystem::Darwin {
67        writeln!(writer, "cargo:rustc-cdylib-link-arg=-undefined").unwrap();
68        writeln!(writer, "cargo:rustc-cdylib-link-arg=dynamic_lookup").unwrap();
69    } else if triple == &Triple::from_str("wasm32-unknown-emscripten").unwrap() {
70        writeln!(writer, "cargo:rustc-cdylib-link-arg=-sSIDE_MODULE=2").unwrap();
71        writeln!(writer, "cargo:rustc-cdylib-link-arg=-sWASM_BIGINT").unwrap();
72    }
73}

75/// Loads the configuration determined from the build environment.
76///
77/// Because this will never change in a given compilation run, this is cached in a `once_cell`.
78#[cfg(feature = "resolve-config")]
79pub fn get() -> &'static InterpreterConfig {
80    static CONFIG: OnceCell<InterpreterConfig> = OnceCell::new();
81    CONFIG.get_or_init(|| {
82        // Check if we are in a build script and cross compiling to a different target.
83        let cross_compile_config_path = resolve_cross_compile_config_path();
84        let cross_compiling = cross_compile_config_path
85            .as_ref()
86            .map(|path| path.exists())
87            .unwrap_or(false);
88
89        if let Some(interpreter_config) = InterpreterConfig::from_cargo_dep_env() {
90            interpreter_config
91        } else if !CONFIG_FILE.is_empty() {
92            InterpreterConfig::from_reader(Cursor::new(CONFIG_FILE))
93        } else if cross_compiling {
94            InterpreterConfig::from_path(cross_compile_config_path.as_ref().unwrap())
95        } else {
96            InterpreterConfig::from_reader(Cursor::new(HOST_CONFIG))
97        }
98        .expect("failed to parse PyO3 config")
99    })
100}

102/// Build configuration provided by `PYO3_CONFIG_FILE`. May be empty if env var not set.
103#[doc(hidden)]
104#[cfg(feature = "resolve-config")]
105const CONFIG_FILE: &str = include_str!(concat!(env!("OUT_DIR"), "/pyo3-build-config-file.txt"));
106
107/// Build configuration discovered by `pyo3-build-config` build script. Not aware of
108/// cross-compilation settings.
109#[doc(hidden)]
110#[cfg(feature = "resolve-config")]
111const HOST_CONFIG: &str = include_str!(concat!(env!("OUT_DIR"), "/pyo3-build-config.txt"));
112
113/// Returns the path where PyO3's build.rs writes its cross compile configuration.
114///
115/// The config file will be named `$OUT_DIR/<triple>/pyo3-build-config.txt`.
116///
117/// Must be called from a build script, returns `None` if not.
118#[doc(hidden)]
119#[cfg(feature = "resolve-config")]
120fn resolve_cross_compile_config_path() -> Option<PathBuf> {
121    env::var_os("TARGET").map(|target| {
122        let mut path = PathBuf::from(env!("OUT_DIR"));
123        path.push(Path::new(&target));
124        path.push("pyo3-build-config.txt");
125        path
126    })
127}

129/// Use certain features if we detect the compiler being used supports them.
130///
131/// Features may be removed or added as MSRV gets bumped or new features become available,
132/// so this function is unstable.
133#[doc(hidden)]
134pub fn print_feature_cfgs() {
135    fn rustc_minor_version() -> Option<u32> {
136        let rustc = env::var_os("RUSTC")?;
137        let output = Command::new(rustc).arg("--version").output().ok()?;
138        let version = core::str::from_utf8(&output.stdout).ok()?;
139        let mut pieces = version.split('.');
140        if pieces.next() != Some("rustc 1") {
141            return None;
142        }
143        pieces.next()?.parse().ok()
144    }
145
146    let rustc_minor_version = rustc_minor_version().unwrap_or(0);
147
148    // Enable use of const initializer for thread_local! on Rust 1.59 and greater
149    if rustc_minor_version >= 59 {
150        println!("cargo:rustc-cfg=thread_local_const_init");
151    }
152
153    // invalid_from_utf8 lint was added in Rust 1.74
154    if rustc_minor_version >= 74 {
155        println!("cargo:rustc-cfg=invalid_from_utf8_lint");
156    }
157}

178    /// Gets the configuration for use from PyO3's build script.
179    ///
180    /// Differs from .get() above only in the cross-compile case, where PyO3's build script is
181    /// required to generate a new config (as it's the first build script which has access to the
182    /// correct value for CARGO_CFG_TARGET_OS).
183    #[cfg(feature = "resolve-config")]
184    pub fn resolve_interpreter_config() -> Result<InterpreterConfig> {
185        if !CONFIG_FILE.is_empty() {
186            let mut interperter_config = InterpreterConfig::from_reader(Cursor::new(CONFIG_FILE))?;
187            interperter_config.generate_import_libs()?;
188            Ok(interperter_config)
189        } else if let Some(interpreter_config) = make_cross_compile_config()? {
190            // This is a cross compile and need to write the config file.
191            let path = resolve_cross_compile_config_path()
192                .expect("resolve_interpreter_config() must be called from a build script");
193            let parent_dir = path.parent().ok_or_else(|| {
194                format!(
195                    "failed to resolve parent directory of config file {}",
196                    path.display()
197                )
198            })?;
199            std::fs::create_dir_all(parent_dir).with_context(|| {
200                format!(
201                    "failed to create config file directory {}",
202                    parent_dir.display()
203                )
204            })?;
205            interpreter_config.to_writer(&mut std::fs::File::create(&path).with_context(
206                || format!("failed to create config file at {}", path.display()),
207            )?)?;
208            Ok(interpreter_config)
209        } else {
210            InterpreterConfig::from_reader(Cursor::new(HOST_CONFIG))
211        }
212    }

215#[cfg(test)]
216mod tests {
217    use super::*;
218
219    #[test]
220    fn extension_module_link_args() {
221        let mut buf = Vec::new();
222
223        // Does nothing on non-mac
224        _add_extension_module_link_args(
225            &Triple::from_str("x86_64-pc-windows-msvc").unwrap(),
226            &mut buf,
227        );
228        assert_eq!(buf, Vec::new());
229
230        _add_extension_module_link_args(
231            &Triple::from_str("x86_64-apple-darwin").unwrap(),
232            &mut buf,
233        );
234        assert_eq!(
235            std::str::from_utf8(&buf).unwrap(),
236            "cargo:rustc-cdylib-link-arg=-undefined\n\
237             cargo:rustc-cdylib-link-arg=dynamic_lookup\n"
238        );
239
240        buf.clear();
241        _add_extension_module_link_args(
242            &Triple::from_str("wasm32-unknown-emscripten").unwrap(),
243            &mut buf,
244        );
245        assert_eq!(
246            std::str::from_utf8(&buf).unwrap(),
247            "cargo:rustc-cdylib-link-arg=-sSIDE_MODULE=2\n\
248             cargo:rustc-cdylib-link-arg=-sWASM_BIGINT\n"
249        );
250    }
