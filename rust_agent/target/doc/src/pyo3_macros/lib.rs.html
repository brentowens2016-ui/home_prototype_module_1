15/// A proc macro used to implement Python modules.
16///
17/// The name of the module will be taken from the function name, unless `#[pyo3(name = "my_name")]`
18/// is also annotated on the function to override the name. **Important**: the module name should
19/// match the `lib.name` setting in `Cargo.toml`, so that Python is able to import the module
20/// without needing a custom import loader.
21///
22/// Functions annotated with `#[pymodule]` can also be annotated with the following:
23///
24/// |  Annotation  |  Description |
25/// | :-  | :- |
26/// | `#[pyo3(name = "...")]` | Defines the name of the module in Python. |
27///
28/// For more on creating Python modules see the [module section of the guide][1].
29///
30/// Due to technical limitations on how `#[pymodule]` is implemented, a function marked
31/// `#[pymodule]` cannot have a module with the same name in the same scope. (The
32/// `#[pymodule]` implementation generates a hidden module with the same name containing
33/// metadata about the module, which is used by `wrap_pymodule!`).
34///
35/// [1]: https://pyo3.rs/latest/module.html
36#[proc_macro_attribute]
37pub fn pymodule(args: TokenStream, input: TokenStream) -> TokenStream {
38    parse_macro_input!(args as Nothing);
39    match parse_macro_input!(input as Item) {
40        Item::Mod(module) => if cfg!(feature = "experimental-declarative-modules") {
41            pymodule_module_impl(module)
42        } else {
43            Err(syn::Error::new_spanned(
44                module,
45                "#[pymodule] requires the 'experimental-declarative-modules' feature to be used on Rust modules.",
46            ))
47        },
48        Item::Fn(function) => pymodule_function_impl(function),
49        unsupported => Err(syn::Error::new_spanned(
50            unsupported,
51            "#[pymodule] only supports modules and functions.",
52        )),
53    }
54    .unwrap_or_compile_error()
55    .into()
56}

58#[proc_macro_attribute]
59pub fn pyclass(attr: TokenStream, input: TokenStream) -> TokenStream {
60    let item = parse_macro_input!(input as Item);
61    match item {
62        Item::Struct(struct_) => pyclass_impl(attr, struct_, methods_type()),
63        Item::Enum(enum_) => pyclass_enum_impl(attr, enum_, methods_type()),
64        unsupported => {
65            syn::Error::new_spanned(unsupported, "#[pyclass] only supports structs and enums.")
66                .into_compile_error()
67                .into()
68        }
69    }
70}

72/// A proc macro used to expose methods to Python.
73///
74/// Methods within a `#[pymethods]` block can be annotated with  as well as the following:
75///
76/// |  Annotation  |  Description |
77/// | :-  | :- |
78/// | [`#[new]`][4]  | Defines the class constructor, like Python's `__new__` method. |
79/// | [`#[getter]`][5] and [`#[setter]`][5] | These define getters and setters, similar to Python's `@property` decorator. This is useful for getters/setters that require computation or side effects; if that is not the case consider using [`#[pyo3(get, set)]`][11] on the struct's field(s).|
80/// | [`#[staticmethod]`][6]| Defines the method as a staticmethod, like Python's `@staticmethod` decorator.|
81/// | [`#[classmethod]`][7]  | Defines the method as a classmethod, like Python's `@classmethod` decorator.|
82/// | [`#[classattr]`][9]  | Defines a class variable. |
83/// | [`#[args]`][10]  | Deprecated way to define a method's default arguments and allows the function to receive `*args` and `**kwargs`. Use `#[pyo3(signature = (...))]` instead. |
84/// | <nobr>[`#[pyo3(<option> = <value>)`][pyo3-method-options]</nobr> | Any of the `#[pyo3]` options supported on [`macro@pyfunction`]. |
85///
86/// For more on creating class methods,
87/// see the [class section of the guide][1].
88///
89/// If the [`multiple-pymethods`][2] feature is enabled, it is possible to implement
90/// multiple `#[pymethods]` blocks for a single `#[pyclass]`.
91/// This will add a transitive dependency on the [`inventory`][3] crate.
92///
93/// [1]: https://pyo3.rs/latest/class.html#instance-methods
94/// [2]: https://pyo3.rs/latest/features.html#multiple-pymethods
95/// [3]: https://docs.rs/inventory/
96/// [4]: https://pyo3.rs/latest/class.html#constructor
97/// [5]: https://pyo3.rs/latest/class.html#object-properties-using-getter-and-setter
98/// [6]: https://pyo3.rs/latest/class.html#static-methods
99/// [7]: https://pyo3.rs/latest/class.html#class-methods
100/// [8]: https://pyo3.rs/latest/class.html#callable-objects
101/// [9]: https://pyo3.rs/latest/class.html#class-attributes
102/// [10]: https://pyo3.rs/latest/class.html#method-arguments
103/// [11]: https://pyo3.rs/latest/class.html#object-properties-using-pyo3get-set
104#[proc_macro_attribute]
105pub fn pymethods(attr: TokenStream, input: TokenStream) -> TokenStream {
106    let methods_type = if cfg!(feature = "multiple-pymethods") {
107        PyClassMethodsType::Inventory
108    } else {
109        PyClassMethodsType::Specialization
110    };
111    pymethods_impl(attr, input, methods_type)
112}

114/// A proc macro used to expose Rust functions to Python.
115///
116/// Functions annotated with `#[pyfunction]` can also be annotated with the following `#[pyo3]`
117/// options:
118///
119/// |  Annotation  |  Description |
120/// | :-  | :- |
121/// | `#[pyo3(name = "...")]` | Defines the name of the function in Python. |
122/// | `#[pyo3(text_signature = "...")]` | Defines the `__text_signature__` attribute of the function in Python. |
123/// | `#[pyo3(pass_module)]` | Passes the module containing the function as a `&PyModule` first argument to the function. |
124///
125/// For more on exposing functions see the [function section of the guide][1].
126///
127/// Due to technical limitations on how `#[pyfunction]` is implemented, a function marked
128/// `#[pyfunction]` cannot have a module with the same name in the same scope. (The
129/// `#[pyfunction]` implementation generates a hidden module with the same name containing
130/// metadata about the function, which is used by `wrap_pyfunction!`).
131///
132/// [1]: https://pyo3.rs/latest/function.html
133#[proc_macro_attribute]
134pub fn pyfunction(attr: TokenStream, input: TokenStream) -> TokenStream {
135    let mut ast = parse_macro_input!(input as syn::ItemFn);
136    let options = parse_macro_input!(attr as PyFunctionOptions);
137
138    let expanded = build_py_function(&mut ast, options).unwrap_or_compile_error();
139
140    quote!(
141        #ast
142        #expanded
143    )
144    .into()
145}

187fn pymethods_impl(
188    attr: TokenStream,
189    input: TokenStream,
190    methods_type: PyClassMethodsType,
191) -> TokenStream {
192    let mut ast = parse_macro_input!(input as syn::ItemImpl);
193    // Apply all options as a #[pyo3] attribute on the ItemImpl
194    // e.g. #[pymethods(crate = "crate")] impl Foo { }
195    // -> #[pyo3(crate = "crate")] impl Foo { }
196    let attr: TokenStream2 = attr.into();
197    ast.attrs.push(syn::parse_quote!( #[pyo3(#attr)] ));
198    let expanded = build_py_methods(&mut ast, methods_type).unwrap_or_compile_error();
199
200    quote!(
201        #ast
202        #expanded
203    )
204    .into()
205}

207fn methods_type() -> PyClassMethodsType {
208    if cfg!(feature = "multiple-pymethods") {
209        PyClassMethodsType::Inventory
210    } else {
211        PyClassMethodsType::Specialization
212    }
