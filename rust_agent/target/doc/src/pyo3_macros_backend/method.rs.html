32impl<'a> FnArg<'a> {
33    /// Transforms a rust fn arg parsed with syn into a method::FnArg
34    pub fn parse(arg: &'a mut syn::FnArg) -> Result<Self> {
35        match arg {
36            syn::FnArg::Receiver(recv) => {
37                bail_spanned!(recv.span() => "unexpected receiver")
38            } // checked in parse_fn_type
39            syn::FnArg::Typed(cap) => {
40                if let syn::Type::ImplTrait(_) = &*cap.ty {
41                    bail_spanned!(cap.ty.span() => IMPL_TRAIT_ERR);
42                }
43
44                let arg_attrs = PyFunctionArgPyO3Attributes::from_attrs(&mut cap.attrs)?;
45                let ident = match &*cap.pat {
46                    syn::Pat::Ident(syn::PatIdent { ident, .. }) => ident,
47                    other => return Err(handle_argument_error(other)),
48                };
49
50                let is_cancel_handle = arg_attrs.cancel_handle.is_some();
51
52                Ok(FnArg {
53                    name: ident,
54                    ty: &cap.ty,
55                    optional: utils::option_type_argument(&cap.ty),
56                    default: None,
57                    py: utils::is_python(&cap.ty),
58                    attrs: arg_attrs,
59                    is_varargs: false,
60                    is_kwargs: false,
61                    is_cancel_handle,
62                })
63            }
64        }
65    }
66
67    pub fn is_regular(&self) -> bool {
68        !self.py && !self.is_cancel_handle && !self.is_kwargs && !self.is_varargs
69    }
70}

72fn handle_argument_error(pat: &syn::Pat) -> syn::Error {
73    let span = pat.span();
74    let msg = match pat {
75        syn::Pat::Wild(_) => "wildcard argument names are not supported",
76        syn::Pat::Struct(_)
77        | syn::Pat::Tuple(_)
78        | syn::Pat::TupleStruct(_)
79        | syn::Pat::Slice(_) => "destructuring in arguments is not supported",
80        _ => "unsupported argument",
81    };
82    syn::Error::new(span, msg)
83}

98impl FnType {
99    pub fn skip_first_rust_argument_in_python_signature(&self) -> bool {
100        match self {
101            FnType::Getter(_)
102            | FnType::Setter(_)
103            | FnType::Fn(_)
104            | FnType::FnClass(_)
105            | FnType::FnNewClass(_)
106            | FnType::FnModule(_) => true,
107            FnType::FnNew | FnType::FnStatic | FnType::ClassAttribute => false,
108        }
109    }
110
111    pub fn self_arg(
112        &self,
113        cls: Option<&syn::Type>,
114        error_mode: ExtractErrorMode,
115        holders: &mut Holders,
116        ctx: &Ctx,
117    ) -> TokenStream {
118        let Ctx { pyo3_path } = ctx;
119        match self {
120            FnType::Getter(st) | FnType::Setter(st) | FnType::Fn(st) => {
121                let mut receiver = st.receiver(
122                    cls.expect("no class given for Fn with a \"self\" receiver"),
123                    error_mode,
124                    holders,
125                    ctx,
126                );
127                syn::Token![,](Span::call_site()).to_tokens(&mut receiver);
128                receiver
129            }
130            FnType::FnNew | FnType::FnStatic | FnType::ClassAttribute => {
131                quote!()
132            }
133            FnType::FnClass(span) | FnType::FnNewClass(span) => {
134                let py = syn::Ident::new("py", Span::call_site());
135                let slf: Ident = syn::Ident::new("_slf_ref", Span::call_site());
136                let pyo3_path = pyo3_path.to_tokens_spanned(*span);
137                quote_spanned! { *span =>
138                    #[allow(clippy::useless_conversion)]
139                    ::std::convert::Into::into(
140                        #pyo3_path::impl_::pymethods::BoundRef::ref_from_ptr(#py, &*(#slf as *const _ as *const *mut _))
141                            .downcast_unchecked::<#pyo3_path::types::PyType>()
142                    ),
143                }
144            }
145            FnType::FnModule(span) => {
146                let py = syn::Ident::new("py", Span::call_site());
147                let slf: Ident = syn::Ident::new("_slf_ref", Span::call_site());
148                let pyo3_path = pyo3_path.to_tokens_spanned(*span);
149                quote_spanned! { *span =>
150                    #[allow(clippy::useless_conversion)]
151                    ::std::convert::Into::into(
152                        #pyo3_path::impl_::pymethods::BoundRef::ref_from_ptr(#py, &*(#slf as *const _ as *const *mut _))
153                            .downcast_unchecked::<#pyo3_path::types::PyModule>()
154                    ),
155                }
156            }
157        }
158    }

167#[derive(Clone, Copy)]
168pub enum ExtractErrorMode {
169    NotImplemented,
170    Raise,
171}
172
173impl ExtractErrorMode {
174    pub fn handle_error(self, extract: TokenStream, ctx: &Ctx) -> TokenStream {
175        let Ctx { pyo3_path } = ctx;
176        match self {
177            ExtractErrorMode::Raise => quote! { #extract? },
178            ExtractErrorMode::NotImplemented => quote! {
179                match #extract {
180                    ::std::result::Result::Ok(value) => value,
181                    ::std::result::Result::Err(_) => { return #pyo3_path::callback::convert(py, py.NotImplemented()); },
182                }
183            },
184        }
185    }
186}

188impl SelfType {
189    pub fn receiver(
190        &self,
191        cls: &syn::Type,
192        error_mode: ExtractErrorMode,
193        holders: &mut Holders,
194        ctx: &Ctx,
195    ) -> TokenStream {
196        // Due to use of quote_spanned in this function, need to bind these idents to the
197        // main macro callsite.
198        let py = syn::Ident::new("py", Span::call_site());
199        let slf = syn::Ident::new("_slf", Span::call_site());
200        let Ctx { pyo3_path } = ctx;
201        match self {
202            SelfType::Receiver { span, mutable } => {
203                let method = if *mutable {
204                    syn::Ident::new("extract_pyclass_ref_mut", *span)
205                } else {
206                    syn::Ident::new("extract_pyclass_ref", *span)
207                };
208                let holder = holders.push_holder(*span);
209                let pyo3_path = pyo3_path.to_tokens_spanned(*span);
210                error_mode.handle_error(
211                    quote_spanned! { *span =>
212                        #pyo3_path::impl_::extract_argument::#method::<#cls>(
213                            #pyo3_path::impl_::pymethods::BoundRef::ref_from_ptr(#py, &#slf).0,
214                            &mut #holder,
215                        )
216                    },
217                    ctx,
218                )
219            }
220            SelfType::TryFromBoundRef(span) => {
221                let pyo3_path = pyo3_path.to_tokens_spanned(*span);
222                error_mode.handle_error(
223                    quote_spanned! { *span =>
224                        #pyo3_path::impl_::pymethods::BoundRef::ref_from_ptr(#py, &#slf).downcast::<#cls>()
225                            .map_err(::std::convert::Into::<#pyo3_path::PyErr>::into)
226                            .and_then(
227                                #[allow(unknown_lints, clippy::unnecessary_fallible_conversions)]  // In case slf is Py<Self> (unknown_lints can be removed when MSRV is 1.75+)
228                                |bound| ::std::convert::TryFrom::try_from(bound).map_err(::std::convert::Into::into)
229                            )
230
231                    },
232                    ctx
233                )
234            }
235        }
236    }

239/// Determines which CPython calling convention a given FnSpec uses.
240#[derive(Clone, Debug)]
241pub enum CallingConvention {
242    Noargs,   // METH_NOARGS
243    Varargs,  // METH_VARARGS | METH_KEYWORDS
244    Fastcall, // METH_FASTCALL | METH_KEYWORDS (not compatible with `abi3` feature)
245    TpNew,    // special convention for tp_new
246}
247
248impl CallingConvention {
249    /// Determine default calling convention from an argument signature.
250    ///
251    /// Different other slots (tp_call, tp_new) can have other requirements
252    /// and are set manually (see `parse_fn_type` below).
253    pub fn from_signature(signature: &FunctionSignature<'_>) -> Self {
254        if signature.python_signature.has_no_args() {
255            Self::Noargs
256        } else if signature.python_signature.kwargs.is_some() {
257            // for functions that accept **kwargs, always prefer varargs
258            Self::Varargs
259        } else if !is_abi3() {
260            // FIXME: available in the stable ABI since 3.10
261            Self::Fastcall
262        } else {
263            Self::Varargs
264        }
265    }
266}

274    pub python_name: syn::Ident,

276    pub convention: CallingConvention,
277    pub text_signature: Option<TextSignatureAttribute>,

283pub fn parse_method_receiver(arg: &syn::FnArg) -> Result<SelfType> {
284    match arg {
285        syn::FnArg::Receiver(
286            recv @ syn::Receiver {
287                reference: None, ..
288            },
289        ) => {
290            bail_spanned!(recv.span() => RECEIVER_BY_VALUE_ERR);
291        }
292        syn::FnArg::Receiver(recv @ syn::Receiver { mutability, .. }) => Ok(SelfType::Receiver {
293            mutable: mutability.is_some(),
294            span: recv.span(),
295        }),
296        syn::FnArg::Typed(syn::PatType { ty, .. }) => {
297            if let syn::Type::ImplTrait(_) = &**ty {
298                bail_spanned!(ty.span() => IMPL_TRAIT_ERR);
299            }
300            Ok(SelfType::TryFromBoundRef(ty.span()))
301        }
302    }
303}

305impl<'a> FnSpec<'a> {
306    /// Parser function signature and function attributes
307    pub fn parse(
308        // Signature is mutable to remove the `Python` argument.
309        sig: &'a mut syn::Signature,
310        meth_attrs: &mut Vec<syn::Attribute>,
311        options: PyFunctionOptions,
312        ctx: &'a Ctx,
313    ) -> Result<FnSpec<'a>> {
314        let PyFunctionOptions {
315            text_signature,
316            name,
317            signature,
318            ..
319        } = options;
320
321        let mut python_name = name.map(|name| name.value.0);
322        let mut deprecations = Deprecations::new(ctx);
323
324        let fn_type = Self::parse_fn_type(sig, meth_attrs, &mut python_name, &mut deprecations)?;
325        ensure_signatures_on_valid_method(&fn_type, signature.as_ref(), text_signature.as_ref())?;
326
327        let name = &sig.ident;
328        let python_name = python_name.as_ref().unwrap_or(name).unraw();
329
330        let arguments: Vec<_> = sig
331            .inputs
332            .iter_mut()
333            .skip(if fn_type.skip_first_rust_argument_in_python_signature() {
334                1
335            } else {
336                0
337            })
338            .map(FnArg::parse)
339            .collect::<Result<_>>()?;
340
341        let signature = if let Some(signature) = signature {
342            FunctionSignature::from_arguments_and_attribute(arguments, signature)?
343        } else {
344            FunctionSignature::from_arguments(arguments)?
345        };
346
347        let convention = if matches!(fn_type, FnType::FnNew | FnType::FnNewClass(_)) {
348            CallingConvention::TpNew
349        } else {
350            CallingConvention::from_signature(&signature)
351        };
352
353        Ok(FnSpec {
354            tp: fn_type,
355            name,
356            convention,
357            python_name,
358            signature,
359            text_signature,
360            asyncness: sig.asyncness,
361            unsafety: sig.unsafety,
362            deprecations,
363        })
364    }

366    pub fn null_terminated_python_name(&self) -> syn::LitStr {
367        syn::LitStr::new(&format!("{}\0", self.python_name), self.python_name.span())

370    fn parse_fn_type(
371        sig: &syn::Signature,
372        meth_attrs: &mut Vec<syn::Attribute>,
373        python_name: &mut Option<syn::Ident>,
374        deprecations: &mut Deprecations<'_>,
375    ) -> Result<FnType> {
376        let mut method_attributes = parse_method_attributes(meth_attrs, deprecations)?;
377
378        let name = &sig.ident;
379        let parse_receiver = |msg: &'static str| {
380            let first_arg = sig
381                .inputs
382                .first()
383                .ok_or_else(|| err_spanned!(sig.span() => msg))?;
384            parse_method_receiver(first_arg)
385        };
386
387        // strip get_ or set_
388        let strip_fn_name = |prefix: &'static str| {
389            name.unraw()
390                .to_string()
391                .strip_prefix(prefix)
392                .map(|stripped| syn::Ident::new(stripped, name.span()))
393        };
394
395        let mut set_name_to_new = || {
396            if let Some(name) = &python_name {
397                bail_spanned!(name.span() => "`name` not allowed with `#[new]`");
398            }
399            *python_name = Some(syn::Ident::new("__new__", Span::call_site()));
400            Ok(())
401        };
402
403        let fn_type = match method_attributes.as_mut_slice() {
404            [] => FnType::Fn(parse_receiver(
405                "static method needs #[staticmethod] attribute",
406            )?),
407            [MethodTypeAttribute::StaticMethod(_)] => FnType::FnStatic,
408            [MethodTypeAttribute::ClassAttribute(_)] => FnType::ClassAttribute,
409            [MethodTypeAttribute::New(_)] => {
410                set_name_to_new()?;
411                FnType::FnNew
412            }
413            [MethodTypeAttribute::New(_), MethodTypeAttribute::ClassMethod(span)]
414            | [MethodTypeAttribute::ClassMethod(span), MethodTypeAttribute::New(_)] => {
415                set_name_to_new()?;
416                FnType::FnNewClass(*span)
417            }
418            [MethodTypeAttribute::ClassMethod(_)] => {
419                // Add a helpful hint if the classmethod doesn't look like a classmethod
420                let span = match sig.inputs.first() {
421                    // Don't actually bother checking the type of the first argument, the compiler
422                    // will error on incorrect type.
423                    Some(syn::FnArg::Typed(first_arg)) => first_arg.ty.span(),
424                    Some(syn::FnArg::Receiver(_)) | None => bail_spanned!(
425                        sig.paren_token.span.join() => "Expected `&Bound<PyType>` or `Py<PyType>` as the first argument to `#[classmethod]`"
426                    ),
427                };
428                FnType::FnClass(span)
429            }
430            [MethodTypeAttribute::Getter(_, name)] => {
431                if let Some(name) = name.take() {
432                    ensure_spanned!(
433                        python_name.replace(name).is_none(),
434                        python_name.span() => "`name` may only be specified once"
435                    );
436                } else if python_name.is_none() {
437                    // Strip off "get_" prefix if needed
438                    *python_name = strip_fn_name("get_");
439                }
440
441                FnType::Getter(parse_receiver("expected receiver for `#[getter]`")?)
442            }
443            [MethodTypeAttribute::Setter(_, name)] => {
444                if let Some(name) = name.take() {
445                    ensure_spanned!(
446                        python_name.replace(name).is_none(),
447                        python_name.span() => "`name` may only be specified once"
448                    );
449                } else if python_name.is_none() {
450                    // Strip off "set_" prefix if needed
451                    *python_name = strip_fn_name("set_");
452                }
453
454                FnType::Setter(parse_receiver("expected receiver for `#[setter]`")?)
455            }
456            [first, rest @ .., last] => {
457                // Join as many of the spans together as possible
458                let span = rest
459                    .iter()
460                    .fold(first.span(), |s, next| s.join(next.span()).unwrap_or(s));
461                let span = span.join(last.span()).unwrap_or(span);
462                // List all the attributes in the error message
463                let mut msg = format!("`{}` may not be combined with", first);
464                let mut is_first = true;
465                for attr in &*rest {
466                    msg.push_str(&format!(" `{}`", attr));
467                    if is_first {
468                        is_first = false;
469                    } else {
470                        msg.push(',');
471                    }
472                }
473                if !rest.is_empty() {
474                    msg.push_str(" and");
475                }
476                msg.push_str(&format!(" `{}`", last));
477                bail_spanned!(span => msg)
478            }
479        };
480        Ok(fn_type)
481    }

483    /// Return a C wrapper function for this signature.
484    pub fn get_wrapper_function(
485        &self,
486        ident: &proc_macro2::Ident,
487        cls: Option<&syn::Type>,
488        ctx: &Ctx,
489    ) -> Result<TokenStream> {
490        let Ctx { pyo3_path } = ctx;
491        let mut cancel_handle_iter = self
492            .signature
493            .arguments
494            .iter()
495            .filter(|arg| arg.is_cancel_handle);
496        let cancel_handle = cancel_handle_iter.next();
497        if let Some(arg) = cancel_handle {
498            ensure_spanned!(self.asyncness.is_some(), arg.name.span() => "`cancel_handle` attribute can only be used with `async fn`");
499            if let Some(arg2) = cancel_handle_iter.next() {
500                bail_spanned!(arg2.name.span() => "`cancel_handle` may only be specified once");
501            }
502        }
503
504        if self.asyncness.is_some() {
505            ensure_spanned!(
506                cfg!(feature = "experimental-async"),
507                self.asyncness.span() => "async functions are only supported with the `experimental-async` feature"
508            );
509        }
510
511        let rust_call = |args: Vec<TokenStream>, holders: &mut Holders| {
512            let mut self_arg = || self.tp.self_arg(cls, ExtractErrorMode::Raise, holders, ctx);
513
514            let call = if self.asyncness.is_some() {
515                let throw_callback = if cancel_handle.is_some() {
516                    quote! { Some(__throw_callback) }
517                } else {
518                    quote! { None }
519                };
520                let python_name = &self.python_name;
521                let qualname_prefix = match cls {
522                    Some(cls) => quote!(Some(<#cls as #pyo3_path::PyTypeInfo>::NAME)),
523                    None => quote!(None),
524                };
525                let arg_names = (0..args.len())
526                    .map(|i| format_ident!("arg_{}", i))
527                    .collect::<Vec<_>>();
528                let future = match self.tp {
529                    FnType::Fn(SelfType::Receiver { mutable: false, .. }) => {
530                        quote! {{
531                            #(let #arg_names = #args;)*
532                            let __guard = #pyo3_path::impl_::coroutine::RefGuard::<#cls>::new(&#pyo3_path::impl_::pymethods::BoundRef::ref_from_ptr(py, &_slf))?;
533                            async move { function(&__guard, #(#arg_names),*).await }
534                        }}
535                    }
536                    FnType::Fn(SelfType::Receiver { mutable: true, .. }) => {
537                        quote! {{
538                            #(let #arg_names = #args;)*
539                            let mut __guard = #pyo3_path::impl_::coroutine::RefMutGuard::<#cls>::new(&#pyo3_path::impl_::pymethods::BoundRef::ref_from_ptr(py, &_slf))?;
540                            async move { function(&mut __guard, #(#arg_names),*).await }
541                        }}
542                    }
543                    _ => {
544                        let self_arg = self_arg();
545                        if self_arg.is_empty() {
546                            quote! { function(#(#args),*) }
547                        } else {
548                            let self_checker = holders.push_gil_refs_checker(self_arg.span());
549                            quote! {
550                                function(
551                                    // NB #self_arg includes a comma, so none inserted here
552                                    #pyo3_path::impl_::deprecations::inspect_type(#self_arg &#self_checker),
553                                    #(#args),*
554                                )
555                            }
556                        }
557                    }
558                };
559                let mut call = quote! {{
560                    let future = #future;
561                    #pyo3_path::impl_::coroutine::new_coroutine(
562                        #pyo3_path::intern!(py, stringify!(#python_name)),
563                        #qualname_prefix,
564                        #throw_callback,
565                        async move { #pyo3_path::impl_::wrap::OkWrap::wrap(future.await) },
566                    )
567                }};
568                if cancel_handle.is_some() {
569                    call = quote! {{
570                        let __cancel_handle = #pyo3_path::coroutine::CancelHandle::new();
571                        let __throw_callback = __cancel_handle.throw_callback();
572                        #call
573                    }};
574                }
575                call
576            } else {
577                let self_arg = self_arg();
578                if self_arg.is_empty() {
579                    quote! { function(#(#args),*) }
580                } else {
581                    let self_checker = holders.push_gil_refs_checker(self_arg.span());
582                    quote! {
583                        function(
584                            // NB #self_arg includes a comma, so none inserted here
585                            #pyo3_path::impl_::deprecations::inspect_type(#self_arg &#self_checker),
586                            #(#args),*
587                        )
588                    }
589                }
590            };
591            quotes::map_result_into_ptr(quotes::ok_wrap(call, ctx), ctx)
592        };
593
594        let func_name = &self.name;
595        let rust_name = if let Some(cls) = cls {
596            quote!(#cls::#func_name)
597        } else {
598            quote!(#func_name)
599        };
600
601        Ok(match self.convention {
602            CallingConvention::Noargs => {
603                let mut holders = Holders::new();
604                let args = self
605                    .signature
606                    .arguments
607                    .iter()
608                    .map(|arg| {
609                        if arg.py {
610                            quote!(py)
611                        } else if arg.is_cancel_handle {
612                            quote!(__cancel_handle)
613                        } else {
614                            unreachable!()
615                        }
616                    })
617                    .collect();
618                let call = rust_call(args, &mut holders);
619                let check_gil_refs = holders.check_gil_refs();
620                let init_holders = holders.init_holders(ctx);
621
622                quote! {
623                    unsafe fn #ident<'py>(
624                        py: #pyo3_path::Python<'py>,
625                        _slf: *mut #pyo3_path::ffi::PyObject,
626                    ) -> #pyo3_path::PyResult<*mut #pyo3_path::ffi::PyObject> {
627                        let _slf_ref = &_slf;
628                        let function = #rust_name; // Shadow the function name to avoid #3017
629                        #init_holders
630                        let result = #call;
631                        #check_gil_refs
632                        result
633                    }
634                }
635            }
636            CallingConvention::Fastcall => {
637                let mut holders = Holders::new();
638                let (arg_convert, args) = impl_arg_params(self, cls, true, &mut holders, ctx)?;
639                let call = rust_call(args, &mut holders);
640                let init_holders = holders.init_holders(ctx);
641                let check_gil_refs = holders.check_gil_refs();
642
643                quote! {
644                    unsafe fn #ident<'py>(
645                        py: #pyo3_path::Python<'py>,
646                        _slf: *mut #pyo3_path::ffi::PyObject,
647                        _args: *const *mut #pyo3_path::ffi::PyObject,
648                        _nargs: #pyo3_path::ffi::Py_ssize_t,
649                        _kwnames: *mut #pyo3_path::ffi::PyObject
650                    ) -> #pyo3_path::PyResult<*mut #pyo3_path::ffi::PyObject> {
651                        let _slf_ref = &_slf;
652                        let function = #rust_name; // Shadow the function name to avoid #3017
653                        #arg_convert
654                        #init_holders
655                        let result = #call;
656                        #check_gil_refs
657                        result
658                    }
659                }
660            }
661            CallingConvention::Varargs => {
662                let mut holders = Holders::new();
663                let (arg_convert, args) = impl_arg_params(self, cls, false, &mut holders, ctx)?;
664                let call = rust_call(args, &mut holders);
665                let init_holders = holders.init_holders(ctx);
666                let check_gil_refs = holders.check_gil_refs();
667
668                quote! {
669                    unsafe fn #ident<'py>(
670                        py: #pyo3_path::Python<'py>,
671                        _slf: *mut #pyo3_path::ffi::PyObject,
672                        _args: *mut #pyo3_path::ffi::PyObject,
673                        _kwargs: *mut #pyo3_path::ffi::PyObject
674                    ) -> #pyo3_path::PyResult<*mut #pyo3_path::ffi::PyObject> {
675                        let _slf_ref = &_slf;
676                        let function = #rust_name; // Shadow the function name to avoid #3017
677                        #arg_convert
678                        #init_holders
679                        let result = #call;
680                        #check_gil_refs
681                        result
682                    }
683                }
684            }
685            CallingConvention::TpNew => {
686                let mut holders = Holders::new();
687                let (arg_convert, args) = impl_arg_params(self, cls, false, &mut holders, ctx)?;
688                let self_arg = self
689                    .tp
690                    .self_arg(cls, ExtractErrorMode::Raise, &mut holders, ctx);
691                let call = quote! { #rust_name(#self_arg #(#args),*) };
692                let init_holders = holders.init_holders(ctx);
693                let check_gil_refs = holders.check_gil_refs();
694                quote! {
695                    unsafe fn #ident(
696                        py: #pyo3_path::Python<'_>,
697                        _slf: *mut #pyo3_path::ffi::PyTypeObject,
698                        _args: *mut #pyo3_path::ffi::PyObject,
699                        _kwargs: *mut #pyo3_path::ffi::PyObject
700                    ) -> #pyo3_path::PyResult<*mut #pyo3_path::ffi::PyObject> {
701                        use #pyo3_path::callback::IntoPyCallbackOutput;
702                        let _slf_ref = &_slf;
703                        let function = #rust_name; // Shadow the function name to avoid #3017
704                        #arg_convert
705                        #init_holders
706                        let result = #call;
707                        let initializer: #pyo3_path::PyClassInitializer::<#cls> = result.convert(py)?;
708                        #check_gil_refs
709                        #pyo3_path::impl_::pymethods::tp_new_impl(py, initializer, _slf)
710                    }
711                }
712            }
713        })
714    }

716    /// Return a `PyMethodDef` constructor for this function, matching the selected
717    /// calling convention.
718    pub fn get_methoddef(&self, wrapper: impl ToTokens, doc: &PythonDoc, ctx: &Ctx) -> TokenStream {
719        let Ctx { pyo3_path } = ctx;
720        let python_name = self.null_terminated_python_name();
721        match self.convention {
722            CallingConvention::Noargs => quote! {
723                #pyo3_path::impl_::pymethods::PyMethodDef::noargs(
724                    #python_name,
725                    #pyo3_path::impl_::pymethods::PyCFunction({
726                        unsafe extern "C" fn trampoline(
727                            _slf: *mut #pyo3_path::ffi::PyObject,
728                            _args: *mut #pyo3_path::ffi::PyObject,
729                        ) -> *mut #pyo3_path::ffi::PyObject
730                        {
731                            #pyo3_path::impl_::trampoline::noargs(
732                                _slf,
733                                _args,
734                                #wrapper
735                            )
736                        }
737                        trampoline
738                    }),
739                    #doc,
740                )
741            },
742            CallingConvention::Fastcall => quote! {
743                #pyo3_path::impl_::pymethods::PyMethodDef::fastcall_cfunction_with_keywords(
744                    #python_name,
745                    #pyo3_path::impl_::pymethods::PyCFunctionFastWithKeywords({
746                        unsafe extern "C" fn trampoline(
747                            _slf: *mut #pyo3_path::ffi::PyObject,
748                            _args: *const *mut #pyo3_path::ffi::PyObject,
749                            _nargs: #pyo3_path::ffi::Py_ssize_t,
750                            _kwnames: *mut #pyo3_path::ffi::PyObject
751                        ) -> *mut #pyo3_path::ffi::PyObject
752                        {
753                            #pyo3_path::impl_::trampoline::fastcall_with_keywords(
754                                _slf,
755                                _args,
756                                _nargs,
757                                _kwnames,
758                                #wrapper
759                            )
760                        }
761                        trampoline
762                    }),
763                    #doc,
764                )
765            },
766            CallingConvention::Varargs => quote! {
767                #pyo3_path::impl_::pymethods::PyMethodDef::cfunction_with_keywords(
768                    #python_name,
769                    #pyo3_path::impl_::pymethods::PyCFunctionWithKeywords({
770                        unsafe extern "C" fn trampoline(
771                            _slf: *mut #pyo3_path::ffi::PyObject,
772                            _args: *mut #pyo3_path::ffi::PyObject,
773                            _kwargs: *mut #pyo3_path::ffi::PyObject,
774                        ) -> *mut #pyo3_path::ffi::PyObject
775                        {
776                            #pyo3_path::impl_::trampoline::cfunction_with_keywords(
777                                _slf,
778                                _args,
779                                _kwargs,
780                                #wrapper
781                            )
782                        }
783                        trampoline
784                    }),
785                    #doc,
786                )
787            },
788            CallingConvention::TpNew => unreachable!("tp_new cannot get a methoddef"),
789        }
790    }

800    /// Creates the parenthesised arguments list for `__text_signature__` snippet based on this spec's signature
801    /// and/or attributes. Prepend the callable name to make a complete `__text_signature__`.
802    pub fn text_signature_call_signature(&self) -> Option<String> {
803        let self_argument = match &self.tp {
804            // Getters / Setters / ClassAttribute are not callables on the Python side
805            FnType::Getter(_) | FnType::Setter(_) | FnType::ClassAttribute => return None,
806            FnType::Fn(_) => Some("self"),
807            FnType::FnModule(_) => Some("module"),
808            FnType::FnClass(_) | FnType::FnNewClass(_) => Some("cls"),
809            FnType::FnStatic | FnType::FnNew => None,
810        };
811
812        match self.text_signature.as_ref().map(|attr| &attr.value) {
813            Some(TextSignatureAttributeValue::Str(s)) => Some(s.value()),
814            None => Some(self.signature.text_signature(self_argument)),
815            Some(TextSignatureAttributeValue::Disabled(_)) => None,
816        }
817    }
818}

820enum MethodTypeAttribute {
821    New(Span),
822    ClassMethod(Span),
823    StaticMethod(Span),
824    Getter(Span, Option<Ident>),
825    Setter(Span, Option<Ident>),
826    ClassAttribute(Span),
827}
828
829impl MethodTypeAttribute {
830    fn span(&self) -> Span {
831        match self {
832            MethodTypeAttribute::New(span)
833            | MethodTypeAttribute::ClassMethod(span)
834            | MethodTypeAttribute::StaticMethod(span)
835            | MethodTypeAttribute::Getter(span, _)
836            | MethodTypeAttribute::Setter(span, _)
837            | MethodTypeAttribute::ClassAttribute(span) => *span,
838        }
839    }

841    /// Attempts to parse a method type attribute.
842    ///
843    /// If the attribute does not match one of the attribute names, returns `Ok(None)`.
844    ///
845    /// Otherwise will either return a parse error or the attribute.
846    fn parse_if_matching_attribute(
847        attr: &syn::Attribute,
848        deprecations: &mut Deprecations<'_>,
849    ) -> Result<Option<Self>> {
850        fn ensure_no_arguments(meta: &syn::Meta, ident: &str) -> syn::Result<()> {
851            match meta {
852                syn::Meta::Path(_) => Ok(()),
853                syn::Meta::List(l) => bail_spanned!(
854                    l.span() => format!(
855                        "`#[{ident}]` does not take any arguments\n= help: did you mean `#[{ident}] #[pyo3({meta})]`?",
856                        ident = ident,
857                        meta = l.tokens,
858                    )
859                ),
860                syn::Meta::NameValue(nv) => {
861                    bail_spanned!(nv.eq_token.span() => format!(
862                        "`#[{}]` does not take any arguments\n= note: this was previously accepted and ignored",
863                        ident
864                    ))
865                }
866            }
867        }
868
869        fn extract_name(meta: &syn::Meta, ident: &str) -> Result<Option<Ident>> {
870            match meta {
871                syn::Meta::Path(_) => Ok(None),
872                syn::Meta::NameValue(nv) => bail_spanned!(
873                    nv.eq_token.span() => format!("expected `#[{}(name)]` to set the name", ident)
874                ),
875                syn::Meta::List(l) => {
876                    if let Ok(name) = l.parse_args::<syn::Ident>() {
877                        Ok(Some(name))
878                    } else if let Ok(name) = l.parse_args::<syn::LitStr>() {
879                        name.parse().map(Some)
880                    } else {
881                        bail_spanned!(l.tokens.span() => "expected ident or string literal for property name");
882                    }
883                }
884            }
885        }
886
887        let meta = &attr.meta;
888        let path = meta.path();
889
890        if path.is_ident("new") {
891            ensure_no_arguments(meta, "new")?;
892            Ok(Some(MethodTypeAttribute::New(path.span())))
893        } else if path.is_ident("__new__") {
894            let span = path.span();
895            deprecations.push(Deprecation::PyMethodsNewDeprecatedForm, span);
896            ensure_no_arguments(meta, "__new__")?;
897            Ok(Some(MethodTypeAttribute::New(span)))
898        } else if path.is_ident("classmethod") {
899            ensure_no_arguments(meta, "classmethod")?;
900            Ok(Some(MethodTypeAttribute::ClassMethod(path.span())))
901        } else if path.is_ident("staticmethod") {
902            ensure_no_arguments(meta, "staticmethod")?;
903            Ok(Some(MethodTypeAttribute::StaticMethod(path.span())))
904        } else if path.is_ident("classattr") {
905            ensure_no_arguments(meta, "classattr")?;
906            Ok(Some(MethodTypeAttribute::ClassAttribute(path.span())))
907        } else if path.is_ident("getter") {
908            let name = extract_name(meta, "getter")?;
909            Ok(Some(MethodTypeAttribute::Getter(path.span(), name)))
910        } else if path.is_ident("setter") {
911            let name = extract_name(meta, "setter")?;
912            Ok(Some(MethodTypeAttribute::Setter(path.span(), name)))
913        } else {
914            Ok(None)
915        }
916    }
917}
918
919impl Display for MethodTypeAttribute {
920    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
921        match self {
922            MethodTypeAttribute::New(_) => "#[new]".fmt(f),
923            MethodTypeAttribute::ClassMethod(_) => "#[classmethod]".fmt(f),
924            MethodTypeAttribute::StaticMethod(_) => "#[staticmethod]".fmt(f),
925            MethodTypeAttribute::Getter(_, _) => "#[getter]".fmt(f),
926            MethodTypeAttribute::Setter(_, _) => "#[setter]".fmt(f),
927            MethodTypeAttribute::ClassAttribute(_) => "#[classattr]".fmt(f),
928        }
929    }
930}
931
932fn parse_method_attributes(
933    attrs: &mut Vec<syn::Attribute>,
934    deprecations: &mut Deprecations<'_>,
935) -> Result<Vec<MethodTypeAttribute>> {
936    let mut new_attrs = Vec::new();
937    let mut found_attrs = Vec::new();
938
939    for attr in attrs.drain(..) {
940        match MethodTypeAttribute::parse_if_matching_attribute(&attr, deprecations)? {
941            Some(attr) => found_attrs.push(attr),
942            None => new_attrs.push(attr),
943        }
944    }
945
946    *attrs = new_attrs;
947
948    Ok(found_attrs)
949}

956fn ensure_signatures_on_valid_method(
957    fn_type: &FnType,
958    signature: Option<&SignatureAttribute>,
959    text_signature: Option<&TextSignatureAttribute>,
960) -> syn::Result<()> {
961    if let Some(signature) = signature {
962        match fn_type {
963            FnType::Getter(_) => {
964                bail_spanned!(signature.kw.span() => "`signature` not allowed with `getter`")
965            }
966            FnType::Setter(_) => {
967                bail_spanned!(signature.kw.span() => "`signature` not allowed with `setter`")
968            }
969            FnType::ClassAttribute => {
970                bail_spanned!(signature.kw.span() => "`signature` not allowed with `classattr`")
971            }
972            _ => {}
973        }
974    }
975    if let Some(text_signature) = text_signature {
976        match fn_type {
977            FnType::Getter(_) => {
978                bail_spanned!(text_signature.kw.span() => "`text_signature` not allowed with `getter`")
979            }
980            FnType::Setter(_) => {
981                bail_spanned!(text_signature.kw.span() => "`text_signature` not allowed with `setter`")
982            }
983            FnType::ClassAttribute => {
984                bail_spanned!(text_signature.kw.span() => "`text_signature` not allowed with `classattr`")
985            }
986            _ => {}
987        }
988    }
989    Ok(())
