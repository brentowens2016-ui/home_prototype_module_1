19/// The mechanism used to collect `#[pymethods]` into the type object
20#[derive(Copy, Clone)]
21pub enum PyClassMethodsType {
22    Specialization,
23    Inventory,
24}
25
26enum PyImplPyO3Option {
27    Crate(CrateAttribute),
28}
29
30impl Parse for PyImplPyO3Option {
31    fn parse(input: ParseStream<'_>) -> Result<Self> {
32        let lookahead = input.lookahead1();
33        if lookahead.peek(syn::Token![crate]) {
34            input.parse().map(PyImplPyO3Option::Crate)
35        } else {
36            Err(lookahead.error())
37        }
38    }
39}

41#[derive(Default)]
42pub struct PyImplOptions {
43    krate: Option<CrateAttribute>,
44}
45
46impl PyImplOptions {
47    pub fn from_attrs(attrs: &mut Vec<syn::Attribute>) -> Result<Self> {
48        let mut options: PyImplOptions = Default::default();
49
50        for option in take_pyo3_options(attrs)? {
51            match option {
52                PyImplPyO3Option::Crate(path) => options.set_crate(path)?,
53            }
54        }
55
56        Ok(options)
57    }
58
59    fn set_crate(&mut self, path: CrateAttribute) -> Result<()> {
60        ensure_spanned!(
61            self.krate.is_none(),
62            path.span() => "`crate` may only be specified once"
63        );
64
65        self.krate = Some(path);
66        Ok(())
67    }
68}

70pub fn build_py_methods(
71    ast: &mut syn::ItemImpl,
72    methods_type: PyClassMethodsType,
73) -> syn::Result<TokenStream> {
74    if let Some((_, path, _)) = &ast.trait_ {
75        bail_spanned!(path.span() => "#[pymethods] cannot be used on trait impl blocks");
76    } else if ast.generics != Default::default() {
77        bail_spanned!(
78            ast.generics.span() =>
79            "#[pymethods] cannot be used with lifetime parameters or generics"
80        );
81    } else {
82        let options = PyImplOptions::from_attrs(&mut ast.attrs)?;
83        impl_methods(&ast.self_ty, &mut ast.items, methods_type, options)
84    }
85}

87pub fn impl_methods(
88    ty: &syn::Type,
89    impls: &mut [syn::ImplItem],
90    methods_type: PyClassMethodsType,
91    options: PyImplOptions,
92) -> syn::Result<TokenStream> {
93    let ctx = &Ctx::new(&options.krate);
94    let mut trait_impls = Vec::new();
95    let mut proto_impls = Vec::new();
96    let mut methods = Vec::new();
97    let mut associated_methods = Vec::new();
98
99    let mut implemented_proto_fragments = HashSet::new();
100
101    for iimpl in impls {
102        match iimpl {
103            syn::ImplItem::Fn(meth) => {
104                let mut fun_options = PyFunctionOptions::from_attrs(&mut meth.attrs)?;
105                fun_options.krate = fun_options.krate.or_else(|| options.krate.clone());
106                match pymethod::gen_py_method(ty, &mut meth.sig, &mut meth.attrs, fun_options, ctx)?
107                {
108                    GeneratedPyMethod::Method(MethodAndMethodDef {
109                        associated_method,
110                        method_def,
111                    }) => {
112                        let attrs = get_cfg_attributes(&meth.attrs);
113                        associated_methods.push(quote!(#(#attrs)* #associated_method));
114                        methods.push(quote!(#(#attrs)* #method_def));
115                    }
116                    GeneratedPyMethod::SlotTraitImpl(method_name, token_stream) => {
117                        implemented_proto_fragments.insert(method_name);
118                        let attrs = get_cfg_attributes(&meth.attrs);
119                        trait_impls.push(quote!(#(#attrs)* #token_stream));
120                    }
121                    GeneratedPyMethod::Proto(MethodAndSlotDef {
122                        associated_method,
123                        slot_def,
124                    }) => {
125                        let attrs = get_cfg_attributes(&meth.attrs);
126                        proto_impls.push(quote!(#(#attrs)* #slot_def));
127                        associated_methods.push(quote!(#(#attrs)* #associated_method));
128                    }
129                }
130            }
131            syn::ImplItem::Const(konst) => {
132                let attributes = ConstAttributes::from_attrs(&mut konst.attrs, ctx)?;
133                if attributes.is_class_attr {
134                    let spec = ConstSpec {
135                        rust_ident: konst.ident.clone(),
136                        attributes,
137                    };
138                    let attrs = get_cfg_attributes(&konst.attrs);
139                    let MethodAndMethodDef {
140                        associated_method,
141                        method_def,
142                    } = gen_py_const(ty, &spec, ctx);
143                    methods.push(quote!(#(#attrs)* #method_def));
144                    associated_methods.push(quote!(#(#attrs)* #associated_method));
145                    if is_proto_method(&spec.python_name().to_string()) {
146                        // If this is a known protocol method e.g. __contains__, then allow this
147                        // symbol even though it's not an uppercase constant.
148                        konst
149                            .attrs
150                            .push(syn::parse_quote!(#[allow(non_upper_case_globals)]));
151                    }
152                }
153            }
154            syn::ImplItem::Macro(m) => bail_spanned!(
155                m.span() =>
156                "macros cannot be used as items in `#[pymethods]` impl blocks\n\
157                 = note: this was previously accepted and ignored"
158            ),
159            _ => {}
160        }
161    }
162
163    add_shared_proto_slots(ty, &mut proto_impls, implemented_proto_fragments, ctx);
164
165    let ctx = &Ctx::new(&options.krate);
166
167    let items = match methods_type {
168        PyClassMethodsType::Specialization => impl_py_methods(ty, methods, proto_impls, ctx),
169        PyClassMethodsType::Inventory => submit_methods_inventory(ty, methods, proto_impls, ctx),
170    };
171
172    Ok(quote! {
173        #(#trait_impls)*
174
175        #items
176
177        #[doc(hidden)]
178        #[allow(non_snake_case)]
179        impl #ty {
180            #(#associated_methods)*
181        }
182    })
183}

185pub fn gen_py_const(cls: &syn::Type, spec: &ConstSpec<'_>, ctx: &Ctx) -> MethodAndMethodDef {
186    let member = &spec.rust_ident;
187    let wrapper_ident = format_ident!("__pymethod_{}__", member);
188    let deprecations = &spec.attributes.deprecations;
189    let python_name = &spec.null_terminated_python_name();
190    let Ctx { pyo3_path } = ctx;
191
192    let associated_method = quote! {
193        fn #wrapper_ident(py: #pyo3_path::Python<'_>) -> #pyo3_path::PyResult<#pyo3_path::PyObject> {
194            #deprecations
195            ::std::result::Result::Ok(#pyo3_path::IntoPy::into_py(#cls::#member, py))
196        }
197    };
198
199    let method_def = quote! {
200        #pyo3_path::class::PyMethodDefType::ClassAttribute({
201            #pyo3_path::class::PyClassAttributeDef::new(
202                #python_name,
203                #pyo3_path::impl_::pymethods::PyClassAttributeFactory(#cls::#wrapper_ident)
204            )
205        })
206    };
207
208    MethodAndMethodDef {
209        associated_method,
210        method_def,
211    }
212}

214fn impl_py_methods(
215    ty: &syn::Type,
216    methods: Vec<TokenStream>,
217    proto_impls: Vec<TokenStream>,
218    ctx: &Ctx,
219) -> TokenStream {
220    let Ctx { pyo3_path } = ctx;
221    quote! {
222        impl #pyo3_path::impl_::pyclass::PyMethods<#ty>
223            for #pyo3_path::impl_::pyclass::PyClassImplCollector<#ty>
224        {
225            fn py_methods(self) -> &'static #pyo3_path::impl_::pyclass::PyClassItems {
226                static ITEMS: #pyo3_path::impl_::pyclass::PyClassItems = #pyo3_path::impl_::pyclass::PyClassItems {
227                    methods: &[#(#methods),*],
228                    slots: &[#(#proto_impls),*]
229                };
230                &ITEMS
231            }
232        }
233    }
234}

236fn add_shared_proto_slots(
237    ty: &syn::Type,
238    proto_impls: &mut Vec<TokenStream>,
239    mut implemented_proto_fragments: HashSet<String>,
240    ctx: &Ctx,
241) {
242    let Ctx { pyo3_path } = ctx;
243    macro_rules! try_add_shared_slot {
244        ($slot:ident, $($fragments:literal),*) => {{
245            let mut implemented = false;
246            $(implemented |= implemented_proto_fragments.remove($fragments));*;
247            if implemented {
248                proto_impls.push(quote! { #pyo3_path::impl_::pyclass::$slot!(#ty) })
249            }
250        }};
251    }
252
253    try_add_shared_slot!(
254        generate_pyclass_getattro_slot,
255        "__getattribute__",
256        "__getattr__"
257    );
258    try_add_shared_slot!(generate_pyclass_setattr_slot, "__setattr__", "__delattr__");
259    try_add_shared_slot!(generate_pyclass_setdescr_slot, "__set__", "__delete__");
260    try_add_shared_slot!(generate_pyclass_setitem_slot, "__setitem__", "__delitem__");
261    try_add_shared_slot!(generate_pyclass_add_slot, "__add__", "__radd__");
262    try_add_shared_slot!(generate_pyclass_sub_slot, "__sub__", "__rsub__");
263    try_add_shared_slot!(generate_pyclass_mul_slot, "__mul__", "__rmul__");
264    try_add_shared_slot!(generate_pyclass_mod_slot, "__mod__", "__rmod__");
265    try_add_shared_slot!(generate_pyclass_divmod_slot, "__divmod__", "__rdivmod__");
266    try_add_shared_slot!(generate_pyclass_lshift_slot, "__lshift__", "__rlshift__");
267    try_add_shared_slot!(generate_pyclass_rshift_slot, "__rshift__", "__rrshift__");
268    try_add_shared_slot!(generate_pyclass_and_slot, "__and__", "__rand__");
269    try_add_shared_slot!(generate_pyclass_or_slot, "__or__", "__ror__");
270    try_add_shared_slot!(generate_pyclass_xor_slot, "__xor__", "__rxor__");
271    try_add_shared_slot!(generate_pyclass_matmul_slot, "__matmul__", "__rmatmul__");
272    try_add_shared_slot!(generate_pyclass_truediv_slot, "__truediv__", "__rtruediv__");
273    try_add_shared_slot!(
274        generate_pyclass_floordiv_slot,
275        "__floordiv__",
276        "__rfloordiv__"
277    );
278    try_add_shared_slot!(generate_pyclass_pow_slot, "__pow__", "__rpow__");
279    try_add_shared_slot!(
280        generate_pyclass_richcompare_slot,
281        "__lt__",
282        "__le__",
283        "__eq__",
284        "__ne__",
285        "__gt__",
286        "__ge__"
287    );
288
289    // if this assertion trips, a slot fragment has been implemented which has not been added in the
290    // list above
291    assert!(implemented_proto_fragments.is_empty());
292}

294fn submit_methods_inventory(
295    ty: &syn::Type,
296    methods: Vec<TokenStream>,
297    proto_impls: Vec<TokenStream>,
298    ctx: &Ctx,
299) -> TokenStream {
300    let Ctx { pyo3_path } = ctx;
301    quote! {
302        #pyo3_path::inventory::submit! {
303            type Inventory = <#ty as #pyo3_path::impl_::pyclass::PyClassImpl>::Inventory;
304            Inventory::new(#pyo3_path::impl_::pyclass::PyClassItems { methods: &[#(#methods),*], slots: &[#(#proto_impls),*] })
305        }
306    }
307}

309fn get_cfg_attributes(attrs: &[syn::Attribute]) -> Vec<&syn::Attribute> {
310    attrs
311        .iter()
312        .filter(|attr| attr.path().is_ident("cfg"))
313        .collect()
