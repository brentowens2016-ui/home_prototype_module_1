24/// Creates a copy of the first zero-terminated wide string in `buf`.
25/// The copy includes the zero terminator.
26fn from_zero_terminated(buf: &[u16]) -> Vec<u16> {
27    let end = buf.iter().position(|&x| x == 0).unwrap_or(buf.len());
28    buf[..=end].to_vec()
29}

31// Realistically, volume names are probably not longer than 44 characters,
32// but the example in the Microsoft documentation uses MAX_PATH as well.
33// https://learn.microsoft.com/en-us/windows/win32/fileio/displaying-volume-paths
34const VOLUME_NAME_SIZE: usize = MAX_PATH as usize + 1;
35
36const ERROR_NO_MORE_FILES: HRESULT = windows::Win32::Foundation::ERROR_NO_MORE_FILES.to_hresult();
37const ERROR_MORE_DATA: HRESULT = windows::Win32::Foundation::ERROR_MORE_DATA.to_hresult();
38
39/// Returns a list of zero-terminated wide strings containing volume GUID paths.
40/// Volume GUID paths have the form `\\?\{xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx}\`.
41///
42/// Rather confusingly, the Win32 API _also_ calls these "volume names".
43pub(crate) fn get_volume_guid_paths() -> Vec<Vec<u16>> {
44    let mut volume_names = Vec::new();
45    unsafe {
46        let mut buf = Box::new([0u16; VOLUME_NAME_SIZE]);
47        let Ok(handle) = FindFirstVolumeW(&mut buf[..]) else {
48            sysinfo_debug!(
49                "Error: FindFirstVolumeW() = {:?}",
50                Error::from_win32().code()
51            );
52            return Vec::new();
53        };
54        volume_names.push(from_zero_terminated(&buf[..]));
55        loop {
56            if FindNextVolumeW(handle, &mut buf[..]).is_err() {
57                if Error::from_win32().code() != ERROR_NO_MORE_FILES {
58                    sysinfo_debug!("Error: FindNextVolumeW = {}", Error::from_win32().code());
59                }
60                break;
61            }
62            volume_names.push(from_zero_terminated(&buf[..]));
63        }
64        if FindVolumeClose(handle).is_err() {
65            sysinfo_debug!("Error: FindVolumeClose = {:?}", Error::from_win32().code());
66        };
67    }
68    volume_names
69}

71/// Given a volume GUID path (`\\?\{xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx}\`), returns all
72/// volume paths (drive letters and mount paths) associated with it
73/// as zero terminated wide strings.
74///
75/// # Safety
76/// `volume_name` must contain a zero-terminated wide string.
77pub(crate) unsafe fn get_volume_path_names_for_volume_name(
78    volume_guid_path: &[u16],
79) -> Vec<Vec<u16>> {
80    let volume_guid_path = PCWSTR::from_raw(volume_guid_path.as_ptr());
81
82    // Initial buffer size is just a guess. There is no direct connection between MAX_PATH
83    // the output of GetVolumePathNamesForVolumeNameW.
84    let mut path_names_buf = vec![0u16; MAX_PATH as usize];
85    let mut path_names_output_size = 0u32;
86    for _ in 0..10 {
87        let volume_path_names = GetVolumePathNamesForVolumeNameW(
88            volume_guid_path,
89            Some(path_names_buf.as_mut_slice()),
90            &mut path_names_output_size,
91        );
92        let code = volume_path_names.map_err(|_| Error::from_win32().code());
93        match code {
94            Ok(()) => break,
95            Err(ERROR_MORE_DATA) => {
96                // We need a bigger buffer. path_names_output_size contains the required buffer size.
97                path_names_buf = vec![0u16; path_names_output_size as usize];
98                continue;
99            }
100            Err(_e) => {
101                sysinfo_debug!("Error: GetVolumePathNamesForVolumeNameW() = {}", _e);
102                return Vec::new();
103            }
104        }
105    }
106
107    // path_names_buf contains multiple zero terminated wide strings.
108    // An additional zero terminates the list.
109    let mut path_names = Vec::new();
110    let mut buf = &path_names_buf[..];
111    while !buf.is_empty() && buf[0] != 0 {
112        let path = from_zero_terminated(buf);
113        buf = &buf[path.len()..];
114        path_names.push(path);
115    }
116    path_names
117}

145    }

150
151    pub(crate) fn available_space(&self) -> u64 {
152        self.available_space
153    }
154
155    pub(crate) fn is_removable(&self) -> bool {
156        self.is_removable
157    }
158
159    pub(crate) fn refresh(&mut self) -> bool {
160        if self.total_space != 0 {
161            unsafe {
162                let mut tmp = 0;
163                let lpdirectoryname = PCWSTR::from_raw(self.mount_point.as_ptr());
164                if GetDiskFreeSpaceExW(lpdirectoryname, None, None, Some(&mut tmp)).is_ok() {
165                    self.available_space = tmp;
166                    return true;
167                }
168            }
169        }
170        false
171    }
172}

208unsafe fn get_drive_size(mount_point: &[u16]) -> Option<(u64, u64)> {
209    let mut total_size = 0;
210    let mut available_space = 0;
211    let lpdirectoryname = PCWSTR::from_raw(mount_point.as_ptr());
212    if GetDiskFreeSpaceExW(
213        lpdirectoryname,
214        None,
215        Some(&mut total_size),
216        Some(&mut available_space),
217    )
218    .is_ok()
219    {
220        Some((total_size, available_space))
221    } else {
222        None
223    }
224}

226pub(crate) unsafe fn get_list() -> Vec<Disk> {
227    #[cfg(feature = "multithread")]
228    use rayon::iter::ParallelIterator;
229
230    crate::utils::into_iter(get_volume_guid_paths())
231        .flat_map(|volume_name| {
232            let raw_volume_name = PCWSTR::from_raw(volume_name.as_ptr());
233            let drive_type = GetDriveTypeW(raw_volume_name);
234
235            let is_removable = drive_type == DRIVE_REMOVABLE;
236
237            if drive_type != DRIVE_FIXED && drive_type != DRIVE_REMOVABLE {
238                return Vec::new();
239            }
240            let mut name = [0u16; MAX_PATH as usize + 1];
241            let mut file_system = [0u16; 32];
242            let volume_info_res = GetVolumeInformationW(
243                raw_volume_name,
244                Some(&mut name),
245                None,
246                None,
247                None,
248                Some(&mut file_system),
249            )
250            .is_ok();
251            if !volume_info_res {
252                sysinfo_debug!(
253                    "Error: GetVolumeInformationW = {:?}",
254                    Error::from_win32().code()
255                );
256                return Vec::new();
257            }
258
259            let mount_paths = get_volume_path_names_for_volume_name(&volume_name[..]);
260            if mount_paths.is_empty() {
261                return Vec::new();
262            }
263
264            // The device path is the volume name without the trailing backslash.
265            let device_path = volume_name[..(volume_name.len() - 2)]
266                .iter()
267                .copied()
268                .chain([0])
269                .collect::<Vec<_>>();
270            let Some(handle) = HandleWrapper::new_from_file(&device_path[..], Default::default())
271            else {
272                return Vec::new();
273            };
274            let Some((total_space, available_space)) = get_drive_size(&mount_paths[0][..]) else {
275                return Vec::new();
276            };
277            if total_space == 0 {
278                sysinfo_debug!("total_space == 0");
279                return Vec::new();
280            }
281            let spq_trim = STORAGE_PROPERTY_QUERY {
282                PropertyId: StorageDeviceSeekPenaltyProperty,
283                QueryType: PropertyStandardQuery,
284                AdditionalParameters: [0],
285            };
286            let mut result: DEVICE_SEEK_PENALTY_DESCRIPTOR = std::mem::zeroed();
287
288            let mut dw_size = 0;
289            let device_io_control = DeviceIoControl(
290                handle.0,
291                IOCTL_STORAGE_QUERY_PROPERTY,
292                Some(&spq_trim as *const STORAGE_PROPERTY_QUERY as *const c_void),
293                size_of::<STORAGE_PROPERTY_QUERY>() as u32,
294                Some(&mut result as *mut DEVICE_SEEK_PENALTY_DESCRIPTOR as *mut c_void),
295                size_of::<DEVICE_SEEK_PENALTY_DESCRIPTOR>() as u32,
296                Some(&mut dw_size),
297                None,
298            )
299            .is_ok();
300            let type_ = if !device_io_control
301                || dw_size != size_of::<DEVICE_SEEK_PENALTY_DESCRIPTOR>() as u32
302            {
303                DiskKind::Unknown(-1)
304            } else {
305                let is_hdd = result.IncursSeekPenalty.as_bool();
306                if is_hdd {
307                    DiskKind::HDD
308                } else {
309                    DiskKind::SSD
310                }
311            };
312
313            let name = os_string_from_zero_terminated(&name);
314            let file_system = os_string_from_zero_terminated(&file_system);
315            mount_paths
316                .into_iter()
317                .map(move |mount_path| Disk {
318                    inner: DiskInner {
319                        type_,
320                        name: name.clone(),
321                        file_system: file_system.clone(),
322                        s_mount_point: OsString::from_wide(&mount_path[..mount_path.len() - 1]),
323                        mount_point: mount_path,
324                        total_space,
325                        available_space,
326                        is_removable,
327                    },
328                })
329                .collect::<Vec<_>>()
330        })
331        .collect::<Vec<_>>()
332}

334fn os_string_from_zero_terminated(name: &[u16]) -> OsString {
335    let len = name.iter().position(|&x| x == 0).unwrap_or(name.len());
336    OsString::from_wide(&name[..len])
