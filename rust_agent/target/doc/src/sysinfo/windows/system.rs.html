51unsafe fn boot_time() -> u64 {
52    match SystemTime::now().duration_since(SystemTime::UNIX_EPOCH) {
53        Ok(n) => n.as_secs().saturating_sub(GetTickCount64() / 1_000),
54        Err(_e) => {
55            sysinfo_debug!("Failed to compute boot time: {:?}", _e);
56            0
57        }
58    }
59}

84    pub(crate) fn refresh_cpu_specifics(&mut self, refresh_kind: CpuRefreshKind) {
85        if self.query.is_none() {
86            self.query = Query::new();
87            if let Some(ref mut query) = self.query {
88                add_english_counter(
89                    r"\Processor(_Total)\% Idle Time".to_string(),
90                    query,
91                    get_key_used(self.cpus.global_cpu_mut()),
92                    "tot_0".to_owned(),
93                );
94                for (pos, proc_) in self.cpus.iter_mut(refresh_kind).enumerate() {
95                    add_english_counter(
96                        format!(r"\Processor({pos})\% Idle Time"),
97                        query,
98                        get_key_used(proc_),
99                        format!("{pos}_0"),
100                    );
101                }
102            }
103        }
104        if let Some(ref mut query) = self.query {
105            query.refresh();
106            let mut total_idle_time = None;
107            if let Some(ref key_used) = *get_key_used(self.cpus.global_cpu_mut()) {
108                total_idle_time = Some(
109                    query
110                        .get(&key_used.unique_id)
111                        .expect("global_key_idle disappeared"),
112                );
113            }
114            if let Some(total_idle_time) = total_idle_time {
115                self.cpus
116                    .global_cpu_mut()
117                    .inner
118                    .set_cpu_usage(100.0 - total_idle_time);
119            }
120            for p in self.cpus.iter_mut(refresh_kind) {
121                let mut idle_time = None;
122                if let Some(ref key_used) = *get_key_used(p) {
123                    idle_time = Some(
124                        query
125                            .get(&key_used.unique_id)
126                            .expect("key_used disappeared"),
127                    );
128                }
129                if let Some(idle_time) = idle_time {
130                    p.inner.set_cpu_usage(100.0 - idle_time);
131                }
132            }
133            if refresh_kind.frequency() {
134                self.cpus.get_frequencies();
135            }
136        }
137    }

139    pub(crate) fn refresh_memory_specifics(&mut self, refresh_kind: MemoryRefreshKind) {
140        unsafe {
141            if refresh_kind.ram() {
142                let mut mem_info: MEMORYSTATUSEX = zeroed();
143                mem_info.dwLength = size_of::<MEMORYSTATUSEX>() as _;
144                let _err = GlobalMemoryStatusEx(&mut mem_info);
145                self.mem_total = mem_info.ullTotalPhys as _;
146                self.mem_available = mem_info.ullAvailPhys as _;
147            }
148            if refresh_kind.swap() {
149                let mut perf_info: PERFORMANCE_INFORMATION = zeroed();
150                if K32GetPerformanceInfo(&mut perf_info, size_of::<PERFORMANCE_INFORMATION>() as _)
151                    .as_bool()
152                {
153                    let page_size = perf_info.PageSize as u64;
154                    let physical_total = perf_info.PhysicalTotal as u64;
155                    let commit_limit = perf_info.CommitLimit as u64;
156                    let commit_total = perf_info.CommitTotal as u64;
157                    self.swap_total =
158                        page_size.saturating_mul(commit_limit.saturating_sub(physical_total));
159                    self.swap_used =
160                        page_size.saturating_mul(commit_total.saturating_sub(physical_total));
161                }
162            }
163        }
164    }

170    #[allow(clippy::map_entry)]
171    pub(crate) fn refresh_process_specifics(
172        &mut self,
173        pid: Pid,
174        refresh_kind: ProcessRefreshKind,
175    ) -> bool {
176        let now = get_now();
177        let nb_cpus = self.cpus.len() as u64;
178
179        if let Some(proc_) = self.process_list.get_mut(&pid) {
180            if let Some(ret) = refresh_existing_process(proc_, nb_cpus, now, refresh_kind) {
181                return ret;
182            }
183            // We need to re-make the process because the PID owner changed.
184        }
185        if let Some(mut p) = ProcessInner::new_from_pid(pid, now) {
186            p.update(refresh_kind, nb_cpus, now, true);
187            p.updated = false;
188            self.process_list.insert(pid, Process { inner: p });
189            true
190        } else {
191            false
192        }
193    }

195    #[allow(clippy::cast_ptr_alignment)]
196    pub(crate) fn refresh_processes_specifics(
197        &mut self,
198        filter: Option<&[Pid]>,
199        refresh_kind: ProcessRefreshKind,
200    ) {
201        // Windows 10 notebook requires at least 512KiB of memory to make it in one go
202        let mut buffer_size = 512 * 1024;
203        let mut process_information: Vec<u8> = Vec::with_capacity(buffer_size);
204
205        unsafe {
206            loop {
207                let mut cb_needed = 0;
208                // reserve(n) ensures the Vec has capacity for n elements on top of len
209                // so we should reserve buffer_size - len. len will always be zero at this point
210                // this is a no-op on the first call as buffer_size == capacity
211                process_information.reserve(buffer_size);
212
213                match NtQuerySystemInformation(
214                    SystemProcessInformation,
215                    process_information.as_mut_ptr() as *mut _,
216                    buffer_size as _,
217                    &mut cb_needed,
218                )
219                .ok()
220                {
221                    Ok(()) => break,
222                    Err(err) if err.code() == STATUS_INFO_LENGTH_MISMATCH.to_hresult() => {
223                        // GetNewBufferSize
224                        if cb_needed == 0 {
225                            buffer_size *= 2;
226                            continue;
227                        }
228                        // allocating a few more kilo bytes just in case there are some new process
229                        // kicked in since new call to NtQuerySystemInformation
230                        buffer_size = (cb_needed + (1024 * 10)) as usize;
231                        continue;
232                    }
233                    Err(_err) => {
234                        sysinfo_debug!(
235                            "Couldn't get process infos: NtQuerySystemInformation returned {}",
236                            _err,
237                        );
238                        return;
239                    }
240                }
241            }
242
243            #[inline(always)]
244            fn real_filter(e: Pid, filter: &[Pid]) -> bool {
245                filter.contains(&e)
246            }
247
248            #[inline(always)]
249            fn empty_filter(_e: Pid, _filter: &[Pid]) -> bool {
250                true
251            }
252
253            #[allow(clippy::type_complexity)]
254            let (filter, filter_callback): (
255                &[Pid],
256                &(dyn Fn(Pid, &[Pid]) -> bool + Sync + Send),
257            ) = if let Some(filter) = filter {
258                (filter, &real_filter)
259            } else {
260                (&[], &empty_filter)
261            };
262
263            // If we reach this point NtQuerySystemInformation succeeded
264            // and the buffer contents are initialized
265            process_information.set_len(buffer_size);
266
267            // Parse the data block to get process information
268            let mut process_ids = Vec::with_capacity(500);
269            let mut process_information_offset = 0;
270            loop {
271                let p = process_information
272                    .as_ptr()
273                    .offset(process_information_offset)
274                    as *const SYSTEM_PROCESS_INFORMATION;
275
276                // read_unaligned is necessary to avoid
277                // misaligned pointer dereference: address must be a multiple of 0x8 but is 0x...
278                // under x86_64 wine (and possibly other systems)
279                let pi = ptr::read_unaligned(p);
280
281                if filter_callback(Pid(pi.UniqueProcessId as _), filter) {
282                    process_ids.push(Wrap(p));
283                }
284
285                if pi.NextEntryOffset == 0 {
286                    break;
287                }
288
289                process_information_offset += pi.NextEntryOffset as isize;
290            }
291            let process_list = Wrap(UnsafeCell::new(&mut self.process_list));
292            let nb_cpus = if refresh_kind.cpu() {
293                self.cpus.len() as u64
294            } else {
295                0
296            };
297
298            let now = get_now();
299
300            #[cfg(feature = "multithread")]
301            use rayon::iter::ParallelIterator;
302
303            // TODO: instead of using parallel iterator only here, would be better to be
304            //       able to run it over `process_information` directly!
305            let processes = into_iter(process_ids)
306                .filter_map(|pi| {
307                    // as above, read_unaligned is necessary
308                    let pi = ptr::read_unaligned(pi.0);
309                    let pid = Pid(pi.UniqueProcessId as _);
310                    let ppid: usize = pi.InheritedFromUniqueProcessId as _;
311                    let parent = if ppid != 0 {
312                        Some(Pid(pi.InheritedFromUniqueProcessId as _))
313                    } else {
314                        None
315                    };
316                    if let Some(proc_) = (*process_list.0.get()).get_mut(&pid) {
317                        let proc_ = &mut proc_.inner;
318                        if proc_
319                            .get_start_time()
320                            .map(|start| start == proc_.start_time())
321                            .unwrap_or(true)
322                        {
323                            if refresh_kind.memory() {
324                                proc_.memory = pi.WorkingSetSize as _;
325                                proc_.virtual_memory = pi.VirtualSize as _;
326                            }
327                            proc_.update(refresh_kind, nb_cpus, now, false);
328                            // Update the parent in case it changed.
329                            proc_.parent = parent;
330                            return None;
331                        }
332                        // If the PID owner changed, we need to recompute the whole process.
333                        sysinfo_debug!("owner changed for PID {}", proc_.pid());
334                    }
335                    let name = get_process_name(&pi, pid);
336                    let (memory, virtual_memory) = if refresh_kind.memory() {
337                        (pi.WorkingSetSize as _, pi.VirtualSize as _)
338                    } else {
339                        (0, 0)
340                    };
341                    let mut p =
342                        ProcessInner::new_full(pid, parent, memory, virtual_memory, name, now);
343                    p.update(refresh_kind.without_memory(), nb_cpus, now, false);
344                    Some(Process { inner: p })
345                })
346                .collect::<Vec<_>>();
347            for p in processes.into_iter() {
348                self.process_list.insert(p.pid(), p);
349            }
350            self.process_list.retain(|_, v| {
351                let x = v.inner.updated;
352                v.inner.updated = false;
353                x
354            });
355        }
356    }

451    pub(crate) fn os_version() -> Option<String> {
452        let build_number = get_reg_string_value(
453            HKEY_LOCAL_MACHINE,
454            "SOFTWARE\\Microsoft\\Windows NT\\CurrentVersion",
455            "CurrentBuildNumber",
456        )
457        .unwrap_or_default();
458        let major = if Self::is_windows_eleven() {
459            11u32
460        } else {
461            u32::from_le_bytes(
462                get_reg_value_u32(
463                    HKEY_LOCAL_MACHINE,
464                    "SOFTWARE\\Microsoft\\Windows NT\\CurrentVersion",
465                    "CurrentMajorVersionNumber",
466                )
467                .unwrap_or_default(),
468            )
469        };
470        Some(format!("{major} ({build_number})"))
471    }

476    pub(crate) fn cpu_arch() -> Option<String> {
477        unsafe {
478            // https://docs.microsoft.com/fr-fr/windows/win32/api/sysinfoapi/ns-sysinfoapi-system_info
479            let mut info = SYSTEM_INFO::default();
480            GetSystemInfo(&mut info);
481            match info.Anonymous.Anonymous.wProcessorArchitecture {
482                SystemInformation::PROCESSOR_ARCHITECTURE_ALPHA => Some("alpha".to_string()),
483                SystemInformation::PROCESSOR_ARCHITECTURE_ALPHA64 => Some("alpha64".to_string()),
484                SystemInformation::PROCESSOR_ARCHITECTURE_AMD64 => Some("x86_64".to_string()),
485                SystemInformation::PROCESSOR_ARCHITECTURE_ARM => Some("arm".to_string()),
486                SystemInformation::PROCESSOR_ARCHITECTURE_ARM32_ON_WIN64 => Some("arm".to_string()),
487                SystemInformation::PROCESSOR_ARCHITECTURE_ARM64 => Some("arm64".to_string()),
488                SystemInformation::PROCESSOR_ARCHITECTURE_IA32_ON_ARM64
489                | SystemInformation::PROCESSOR_ARCHITECTURE_IA32_ON_WIN64 => {
490                    Some("ia32".to_string())
491                }
492                SystemInformation::PROCESSOR_ARCHITECTURE_IA64 => Some("ia64".to_string()),
493                SystemInformation::PROCESSOR_ARCHITECTURE_INTEL => Some("x86".to_string()),
494                SystemInformation::PROCESSOR_ARCHITECTURE_MIPS => Some("mips".to_string()),
495                SystemInformation::PROCESSOR_ARCHITECTURE_PPC => Some("powerpc".to_string()),
496                _ => None,
497            }
498        }
499    }

502pub(crate) fn is_proc_running(handle: HANDLE) -> bool {
503    let mut exit_code = 0;
504    unsafe { GetExitCodeProcess(handle, &mut exit_code) }.is_ok()
505        && exit_code == STILL_ACTIVE.0 as u32
506}

508/// If it returns `None`, it means that the PID owner changed and that the `Process` must be
509/// completely recomputed.
510fn refresh_existing_process(
511    proc_: &mut Process,
512    nb_cpus: u64,
513    now: u64,
514    refresh_kind: ProcessRefreshKind,
515) -> Option<bool> {
516    let proc_ = &mut proc_.inner;
517    if let Some(handle) = proc_.get_handle() {
518        if get_start_time(handle) != proc_.start_time() {
519            sysinfo_debug!("owner changed for PID {}", proc_.pid());
520            // PID owner changed!
521            return None;
522        }
523        if !is_proc_running(handle) {
524            return Some(false);
525        }
526    } else {
527        return Some(false);
528    }
529    proc_.update(refresh_kind, nb_cpus, now, false);
530    proc_.updated = false;
531    Some(true)
532}

534#[allow(clippy::size_of_in_element_count)]
535//^ needed for "name.Length as usize / std::mem::size_of::<u16>()"
536pub(crate) fn get_process_name(process: &SYSTEM_PROCESS_INFORMATION, process_id: Pid) -> String {
537    let name = &process.ImageName;
538    if name.Buffer.is_null() {
539        match process_id.0 {
540            0 => "Idle".to_owned(),
541            4 => "System".to_owned(),
542            _ => format!("<no name> Process {process_id}"),
543        }
544    } else {
545        unsafe {
546            let slice = std::slice::from_raw_parts(
547                name.Buffer,
548                // The length is in bytes, not the length of string
549                name.Length as usize / std::mem::size_of::<u16>(),
550            );
551
552            String::from_utf16_lossy(slice)
553        }
554    }
555}

557fn get_dns_hostname() -> Option<String> {
558    let mut buffer_size = 0;
559    // Running this first to get the buffer size since the DNS name can be longer than MAX_COMPUTERNAME_LENGTH
560    // setting the `lpBuffer` to null will return the buffer size
561    // https://docs.microsoft.com/en-us/windows/win32/api/sysinfoapi/nf-sysinfoapi-getcomputernameexw
562    unsafe {
563        let _err = GetComputerNameExW(
564            ComputerNamePhysicalDnsHostname,
565            PWSTR::null(),
566            &mut buffer_size,
567        );
568
569        // Setting the buffer with the new length
570        let mut buffer = vec![0_u16; buffer_size as usize];
571
572        // https://docs.microsoft.com/en-us/windows/win32/api/sysinfoapi/ne-sysinfoapi-computer_name_format
573        if GetComputerNameExW(
574            ComputerNamePhysicalDnsHostname,
575            PWSTR::from_raw(buffer.as_mut_ptr()),
576            &mut buffer_size,
577        )
578        .is_ok()
579        {
580            if let Some(pos) = buffer.iter().position(|c| *c == 0) {
581                buffer.resize(pos, 0);
582            }
583
584            return String::from_utf16(&buffer).ok();
585        }
586    }
587
588    sysinfo_debug!("Failed to get computer hostname");
589    None
