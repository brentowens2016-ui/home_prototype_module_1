146pub fn CTL_CODE(
147    DeviceType: DWORD,
148    Function: DWORD,
149    Method: DWORD,
150    Access: DWORD,
151) -> DWORD {
152    (DeviceType << 16) | (Access << 14) | (Function << 2) | Method
153}
